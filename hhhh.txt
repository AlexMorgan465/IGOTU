' UserForm Code (Name: frmFileManager)
' Create a new UserForm with this layout:
' ===== ÉTAPE 1 =====
' - Frame1: Caption = "ÉTAPE 1: A TRAITER → EN COURS"
'   - Label1: "Dossier source (A TRAITER):"
'   - TextBox1: txtSourcePathEtape1 (Multiline = False, Width = 400)
'   - CommandButton1: btnBrowseEtape1 (Caption = "Parcourir A TRAITER...")
'   - ListBox1: lstFilesEtape1 (Width = 500, Height = 120)
'   - CommandButton2: btnToEnCours (Caption = "Déplacer vers EN COURS")
'
' ===== ÉTAPE 2 =====
' - Frame2: Caption = "ÉTAPE 2: EN COURS → TRAITES/NON CONFORME"
'   - Label2: "Dossier source (LOCAL):"
'   - TextBox2: txtSourcePathEtape2 (Multiline = False, Width = 400)
'   - CommandButton3: btnBrowseEtape2 (Caption = "Parcourir dossier LOCAL...")
'   - ListBox2: lstFoldersEtape2 (Width = 500, Height = 120)
'   - Frame3: Caption = "Destination"
'     - OptionButton1: optToTraite (Caption = "TRAITES")
'     - OptionButton2: optToNonConforme (Caption = "NON CONFORME")
'   - CommandButton4: btnMoveFinal (Caption = "Déplacer dossier vers destination")
'
' ===== LISTE DE RÉFÉRENCE EN COURS =====
' - Frame4: Caption = "Fichiers en attente dans EN COURS"
'   - ListBox3: lstEnCoursFiles (Width = 500, Height = 100)
'
' ===== LOG =====
' - ListBox4: lstLog (Width = 500, Height = 80)
' - CommandButton5: btnRefresh (Caption = "Rafraîchir tout")

Option Explicit

Private basePath As String
Private currentFolderEtape1 As String
Private currentFolderEtape2 As String
Private selectedFileEtape1 As String
Private selectedFolderEtape2 As String
Private selectedEnCoursItem As String

Private Sub UserForm_Initialize()
    basePath = "\\UF11-a03\eids-eaa-pdf-statiques\Accessibilité\"
    currentFolderEtape1 = basePath
    currentFolderEtape2 = "C:\" ' Dossier local par défaut
    
    txtSourcePathEtape1.Text = currentFolderEtape1
    txtSourcePathEtape2.Text = currentFolderEtape2
    
    LoadFilesEtape1
    LoadEnCoursFiles
    LoadTrackingData
End Sub

' ===== BOUTONS DE BROWSING SÉPARÉS =====

Private Sub btnBrowseEtape1_Click()
    Dim fd As FileDialog
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)
    
    With fd
        .Title = "Sélectionnez un dossier sous A TRAITER"
        .InitialFileName = currentFolderEtape1
        If .Show = -1 Then
            currentFolderEtape1 = .SelectedItems(1) & "\"
            txtSourcePathEtape1.Text = currentFolderEtape1
            LoadFilesEtape1
        End If
    End With
End Sub

Private Sub btnBrowseEtape2_Click()
    Dim fd As FileDialog
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)
    
    With fd
        .Title = "Sélectionnez un dossier LOCAL contenant les fichiers traités"
        .InitialFileName = currentFolderEtape2
        If .Show = -1 Then
            currentFolderEtape2 = .SelectedItems(1) & "\"
            txtSourcePathEtape2.Text = currentFolderEtape2
            LoadFoldersEtape2
        End If
    End With
End Sub

' ===== GESTIONNAIRES D'ÉVÉNEMENTS =====

Private Sub lstFilesEtape1_Change()
    If lstFilesEtape1.ListIndex >= 0 Then
        ' Récupérer le nom du fichier sélectionné
        selectedFileEtape1 = lstFilesEtape1.List(lstFilesEtape1.ListIndex)
    End If
End Sub

Private Sub lstFoldersEtape2_Change()
    If lstFoldersEtape2.ListIndex >= 0 Then
        ' Récupérer le nom du dossier sélectionné (enlever le préfixe [DOSSIER])
        selectedFolderEtape2 = Replace(lstFoldersEtape2.List(lstFoldersEtape2.ListIndex), "[DOSSIER] ", "")
    End If
End Sub

Private Sub lstEnCoursFiles_Change()
    If lstEnCoursFiles.ListIndex >= 0 Then
        Dim itemText As String
        itemText = lstEnCoursFiles.List(lstEnCoursFiles.ListIndex)
        
        ' Extraire le nom du fichier (entre crochets)
        Dim startPos As Integer
        Dim endPos As Integer
        startPos = InStr(itemText, "[") + 1
        endPos = InStr(itemText, "]")
        
        If startPos > 0 And endPos > startPos Then
            selectedEnCoursItem = Mid(itemText, startPos, endPos - startPos)
        End If
    End If
End Sub

' ===== ÉTAPE 1: DÉPLACER FICHIER VERS EN COURS =====

Private Sub btnToEnCours_Click()
    If selectedFileEtape1 = "" Then
        MsgBox "Veuillez sélectionner un fichier d'abord.", vbExclamation
        Exit Sub
    End If
    
    Dim sourcePath As String
    Dim destPath As String
    Dim fileName As String
    Dim folderName As String
    
    ' Chemin source complet
    sourcePath = currentFolderEtape1 & selectedFileEtape1
    
    ' Vérifier que c'est un fichier (pas un dossier)
    If (GetAttr(sourcePath) And vbDirectory) = vbDirectory Then
        MsgBox "Veuillez sélectionner un FICHIER, pas un dossier.", vbExclamation
        Exit Sub
    End If
    
    ' Nom du fichier sans extension (pour le tracking)
    fileName = selectedFileEtape1
    folderName = Left(selectedFileEtape1, InStrRev(selectedFileEtape1, ".") - 1)
    
    ' Extraire le client et le sous-dossier
    Dim clientName As String
    Dim subPath As String
    
    If Not ExtractClientAndSubPath(currentFolderEtape1, "A TRAITER", clientName, subPath) Then
        Exit Sub
    End If
    
    ' Construire le chemin de destination dans EN COURS
    destPath = basePath & clientName & "\EN COURS\" & subPath
    
    ' Créer le dossier de destination si nécessaire
    On Error GoTo CreateFolderError
    If Dir(destPath, vbDirectory) = "" Then
        MkDir destPath
    End If
    
    ' Copier le fichier vers EN COURS
    On Error GoTo MoveError
    FileCopy sourcePath, destPath & selectedFileEtape1
    
    ' Supprimer le fichier source
    Kill sourcePath
    
    ' Enregistrer la première prise en charge
    RecordFirstPick folderName, clientName & "\" & subPath, fileName
    
    ' Log
    lstLog.AddItem Format(Now, "yyyy-mm-dd hh:mm:ss") & " - ÉTAPE 1: " & selectedFileEtape1 & " → EN COURS (" & clientName & ")"
    
    ' Rafraîchir les listes
    LoadFilesEtape1
    LoadEnCoursFiles
    
    MsgBox "Fichier déplacé avec succès vers EN COURS!", vbInformation
    Exit Sub
    
CreateFolderError:
    MsgBox "Erreur création dossier: " & Err.Description, vbCritical
    Exit Sub
    
MoveError:
    MsgBox "Erreur déplacement fichier: " & Err.Description, vbCritical
End Sub

' ===== ÉTAPE 2: DÉPLACER DOSSIER VERS TRAITES/NON CONFORME =====

Private Sub btnMoveFinal_Click()
    If selectedFolderEtape2 = "" Then
        MsgBox "Veuillez sélectionner un dossier LOCAL d'abord.", vbExclamation
        Exit Sub
    End If
    
    If selectedEnCoursItem = "" Then
        MsgBox "Veuillez sélectionner un fichier dans la liste EN COURS qui correspond à votre dossier.", vbExclamation
        Exit Sub
    End If
    
    ' Vérifier que les noms correspondent
    If selectedFolderEtape2 <> selectedEnCoursItem Then
        MsgBox "Le nom du dossier LOCAL doit correspondre au nom du fichier EN COURS." & vbCrLf & _
               "Dossier local: " & selectedFolderEtape2 & vbCrLf & _
               "Fichier EN COURS: " & selectedEnCoursItem, vbExclamation
        Exit Sub
    End If
    
    If Not optToTraite And Not optToNonConforme Then
        MsgBox "Veuillez sélectionner une destination (TRAITES ou NON CONFORME).", vbExclamation
        Exit Sub
    End If
    
    Dim destType As String
    If optToTraite Then
        destType = "TRAITES"
    Else
        destType = "NON CONFORME"
    End If
    
    ' Récupérer les informations du fichier EN COURS depuis le tracking
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    Dim clientName As String
    Dim subPath As String
    Dim fileName As String
    Dim found As Boolean
    
    Dim i As Long
    For i = 2 To lastRow
        If ws.Cells(i, 1).Value = selectedEnCoursItem And IsEmpty(ws.Cells(i, 5).Value) Then
            fileName = ws.Cells(i, 9).Value ' On stocke le nom du fichier original
            Dim fullPath As String
            fullPath = ws.Cells(i, 2).Value
            
            ' Extraire client et subPath
            Dim pathParts() As String
            pathParts = Split(fullPath, "\")
            
            If UBound(pathParts) >= 0 Then
                clientName = pathParts(0)
                If UBound(pathParts) >= 1 Then
                    subPath = Mid(fullPath, Len(clientName) + 2)
                Else
                    subPath = ""
                End If
            End If
            
            found = True
            Exit For
        End If
    Next i
    
    If Not found Then
        MsgBox "Fichier EN COURS non trouvé dans le tracking.", vbExclamation
        Exit Sub
    End If
    
    ' Chemins
    Dim sourceFolder As String
    sourceFolder = currentFolderEtape2 & selectedFolderEtape2 & "\"
    
    Dim enCoursFilePath As String
    enCoursFilePath = basePath & clientName & "\EN COURS\" & subPath & fileName
    
    Dim destFolder As String
    destFolder = basePath & clientName & "\" & destType & "\" & subPath & selectedFolderEtape2 & "\"
    
    ' Vérifications
    If Dir(sourceFolder, vbDirectory) = "" Then
        MsgBox "Dossier local non trouvé: " & sourceFolder, vbExclamation
        Exit Sub
    End If
    
    If Dir(enCoursFilePath) = "" Then
        MsgBox "Fichier EN COURS non trouvé: " & enCoursFilePath, vbExclamation
        Exit Sub
    End If
    
    ' Créer le dossier de destination
    On Error GoTo CreateDestError
    If Dir(destFolder, vbDirectory) = "" Then
        CreateFolderPath destFolder
    End If
    
    ' Déplacer les fichiers du dossier LOCAL vers la destination
    On Error GoTo MoveError
    
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    
    ' Copier tous les fichiers du dossier local vers la destination
    Dim fileInFolder As String
    fileInFolder = Dir(sourceFolder & "*.*")
    
    Do While fileInFolder <> ""
        FileCopy sourceFolder & fileInFolder, destFolder & fileInFolder
        Kill sourceFolder & fileInFolder
        fileInFolder = Dir()
    Loop
    
    ' Supprimer le dossier local vide
    RmDir sourceFolder
    
    ' Supprimer le fichier dans EN COURS
    Kill enCoursFilePath
    
    ' Vérifier si le dossier EN COURS est vide, si oui le supprimer
    Dim enCoursFolder As String
    enCoursFolder = basePath & clientName & "\EN COURS\" & subPath
    If IsFolderEmpty(enCoursFolder) Then
        RmDir enCoursFolder
    End If
    
    ' Enregistrer la deuxième prise en charge
    RecordSecondPick selectedFolderEtape2, clientName & "\" & subPath, destType
    
    ' Log
    lstLog.AddItem Format(Now, "yyyy-mm-dd hh:mm:ss") & " - ÉTAPE 2: " & selectedFolderEtape2 & " → " & destType & " (" & clientName & ")"
    
    ' Rafraîchir
    LoadFoldersEtape2
    LoadEnCoursFiles
    
    MsgBox "Dossier traité avec succès!", vbInformation
    Exit Sub
    
CreateDestError:
    MsgBox "Erreur création dossier destination: " & Err.Description, vbCritical
    Exit Sub
    
MoveError:
    MsgBox "Erreur déplacement: " & Err.Description, vbCritical
End Sub

' ===== FONCTIONS UTILITAIRES =====

Private Function ExtractClientAndSubPath(fullPath As String, statusFolder As String, ByRef clientName As String, ByRef subPath As String) As Boolean
    ' Extraire le client et le sous-dossier à partir du chemin complet
    If InStr(fullPath, basePath) <> 1 Then
        MsgBox "Le chemin doit commencer par: " & basePath, vbExclamation
        ExtractClientAndSubPath = False
        Exit Function
    End If
    
    Dim remainingPath As String
    remainingPath = Mid(fullPath, Len(basePath) + 1)
    
    Dim parts() As String
    parts = Split(remainingPath, "\")
    
    If UBound(parts) >= 0 Then
        clientName = parts(0)
        
        ' Trouver le dossier status (A TRAITER, EN COURS, etc.)
        Dim statusPos As Integer
        statusPos = InStr(remainingPath, statusFolder & "\")
        
        If statusPos > 0 Then
            subPath = Mid(remainingPath, statusPos + Len(statusFolder) + 1)
        Else
            subPath = ""
        End If
        
        ExtractClientAndSubPath = True
    Else
        ExtractClientAndSubPath = False
    End If
End Function

Private Sub CreateFolderPath(folderPath As String)
    Dim parts() As String
    parts = Split(folderPath, "\")
    
    Dim buildPath As String
    buildPath = parts(0) & "\" & parts(1) & "\" & parts(2) & "\" & parts(3) & "\"
    
    Dim i As Integer
    For i = 4 To UBound(parts) - 1
        buildPath = buildPath & parts(i) & "\"
        If Dir(buildPath, vbDirectory) = "" Then
            MkDir buildPath
        End If
    Next i
    
    ' Créer le dossier final
    MkDir folderPath
End Sub

Private Function IsFolderEmpty(folderPath As String) As Boolean
    Dim fileCount As Integer
    fileCount = 0
    
    Dim fileName As String
    fileName = Dir(folderPath & "*.*")
    
    Do While fileName <> ""
        fileCount = fileCount + 1
        fileName = Dir()
    Loop
    
    IsFolderEmpty = (fileCount = 0)
End Function

' ===== CHARGEMENT DES LISTES =====

Private Sub LoadFilesEtape1()
    lstFilesEtape1.Clear
    
    Dim folder As String
    folder = txtSourcePathEtape1.Text
    
    If Dir(folder, vbDirectory) = "" Then
        Exit Sub
    End If
    
    ' Lister uniquement les FICHIERS (pas les dossiers)
    Dim fileName As String
    fileName = Dir(folder & "*.*")
    
    Do While fileName <> ""
        If (GetAttr(folder & fileName) And vbDirectory) <> vbDirectory Then
            lstFilesEtape1.AddItem fileName
        End If
        fileName = Dir()
    Loop
    
    If lstFilesEtape1.ListCount = 0 Then
        lstFilesEtape1.AddItem "Aucun fichier trouvé"
    End If
    
    selectedFileEtape1 = ""
End Sub

Private Sub LoadFoldersEtape2()
    lstFoldersEtape2.Clear
    
    Dim folder As String
    folder = txtSourcePathEtape2.Text
    
    If Dir(folder, vbDirectory) = "" Then
        Exit Sub
    End If
    
    ' Lister uniquement les DOSSIERS
    Dim fileName As String
    fileName = Dir(folder & "*", vbDirectory)
    
    Do While fileName <> ""
        If fileName <> "." And fileName <> ".." Then
            If (GetAttr(folder & fileName) And vbDirectory) = vbDirectory Then
                lstFoldersEtape2.AddItem "[DOSSIER] " & fileName
            End If
        End If
        fileName = Dir()
    Loop
    
    If lstFoldersEtape2.ListCount = 0 Then
        lstFoldersEtape2.AddItem "Aucun dossier trouvé"
    End If
    
    selectedFolderEtape2 = ""
End Sub

Private Sub LoadEnCoursFiles()
    lstEnCoursFiles.Clear
    
    ' Récupérer tous les fichiers actuellement dans EN COURS depuis le tracking
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If lastRow > 1 Then
        Dim i As Long
        For i = 2 To lastRow
            ' Afficher uniquement les éléments en cours (première date remplie, deuxième vide)
            If Not IsEmpty(ws.Cells(i, 3).Value) And IsEmpty(ws.Cells(i, 5).Value) Then
                lstEnCoursFiles.AddItem ws.Cells(i, 1).Value & " [" & ws.Cells(i, 9).Value & "] - " & ws.Cells(i, 2).Value
            End If
        Next i
    End If
    
    If lstEnCoursFiles.ListCount = 0 Then
        lstEnCoursFiles.AddItem "Aucun fichier en cours"
    End If
    
    selectedEnCoursItem = ""
End Sub

Private Sub btnRefresh_Click()
    LoadFilesEtape1
    LoadFoldersEtape2
    LoadEnCoursFiles
    LoadTrackingData
End Sub

' ===== TRACKING =====

Private Sub RecordFirstPick(folderName As String, fullPath As String, fileName As String)
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim nextRow As Long
    nextRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    
    ' Enregistrer
    ws.Cells(nextRow, 1).Value = folderName  ' Nom du dossier (sans extension)
    ws.Cells(nextRow, 2).Value = fullPath    ' Chemin complet
    ws.Cells(nextRow, 3).Value = Now         ' Date première prise en charge
    ws.Cells(nextRow, 4).Value = Environ("username") ' Agent
    ws.Cells(nextRow, 7).Value = ThisWorkbook.Name ' Session
    ws.Cells(nextRow, 9).Value = fileName    ' Nom du fichier original
End Sub

Private Sub RecordSecondPick(folderName As String, fullPath As String, destType As String)
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    ' Trouver l'enregistrement correspondant
    Dim i As Long
    For i = 2 To lastRow
        If ws.Cells(i, 1).Value = folderName And _
           ws.Cells(i, 2).Value = fullPath And _
           IsEmpty(ws.Cells(i, 5).Value) Then
            
            ' Deuxième prise en charge
            ws.Cells(i, 5).Value = Now
            ws.Cells(i, 6).Value = destType
            
            ' Calculer DMT
            Dim dmtMinutes As Double
            dmtMinutes = DateDiff("n", ws.Cells(i, 3).Value, Now)
            ws.Cells(i, 8).Value = dmtMinutes / 1440
            ws.Cells(i, 8).NumberFormat = "[h]:mm:ss"
            
            Exit For
        End If
    Next i
End Sub

Private Function GetTrackingWorksheet() As Worksheet
    Dim ws As Worksheet
    Dim trackingSheetName As String
    trackingSheetName = "Tracking"
    
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(trackingSheetName)
    On Error GoTo 0
    
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = trackingSheetName
        
        ' En-têtes
        With ws
            .Cells(1, 1).Value = "Nom Dossier"
            .Cells(1, 2).Value = "Chemin (Client\Sous-dossier)"
            .Cells(1, 3).Value = "Date Prise en Charge (Étape 1)"
            .Cells(1, 4).Value = "Agent"
            .Cells(1, 5).Value = "Date Fin (Étape 2)"
            .Cells(1, 6).Value = "Destination Finale"
            .Cells(1, 7).Value = "Session/Workbook"
            .Cells(1, 8).Value = "DMT"
            .Cells(1, 9).Value = "Nom Fichier Original"
            
            .Range("A1:I1").Font.Bold = True
            .Range("A1:I1").Interior.Color = RGB(200, 200, 200)
            .Columns("A:I").AutoFit
        End With
    End If
    
    Set GetTrackingWorksheet = ws
End Function

Private Sub LoadTrackingData()
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If lastRow > 1 Then
        lstLog.Clear
        lstLog.AddItem "=== Activité récente ==="
        
        Dim i As Long
        Dim startRow As Long
        startRow = Application.WorksheetFunction.Max(2, lastRow - 9)
        
        For i = lastRow To startRow Step -1
            Dim logEntry As String
            logEntry = Format(ws.Cells(i, 3).Value, "dd/mm/yyyy hh:mm") & " - " & ws.Cells(i, 1).Value & " par " & ws.Cells(i, 4).Value
            
            If Not IsEmpty(ws.Cells(i, 5).Value) Then
                logEntry = logEntry & " → Terminé en " & Format(ws.Cells(i, 8).Value, "[h]:mm:ss")
            Else
                logEntry = logEntry & " → En cours"
            End If
            
            lstLog.AddItem logEntry
        Next i
    End If
End Sub
