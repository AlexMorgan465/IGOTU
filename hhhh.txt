<script>

let scoreFinal = 0
let nbCritere = 0

function analyser(){

const file = document.getElementById("fileInput").files[0]

if(!file){
alert("Importer une grille Excel")
return
}

const reader = new FileReader()

reader.onload = function(e){

const data = new Uint8Array(e.target.result)
const workbook = XLSX.read(data,{type:'array'})
const sheet = workbook.Sheets[workbook.SheetNames[0]]

const json = XLSX.utils.sheet_to_json(sheet)

let score = 0
let scoreMax = 0

json.forEach(row =>{

let status = row.Status || row.status || ""

let bloquant = row.Bloquant || row.bloquant || "Non"

let poids = bloquant === "Oui" ? 5 : 2

scoreMax += poids
nbCritere++

if(status === "Conforme")
score += poids

else if(status === "Partiellement conforme")
score += poids*0.5

})

scoreFinal = Math.round((score/scoreMax)*100)

document.getElementById("nb").innerText = nbCritere
document.getElementById("score").innerText = scoreFinal

document.getElementById("resultat").style.display="block"

}

reader.readAsArrayBuffer(file)

}



function exportExcel(){

const jour = document.getElementById("jour").value
const semaine = document.getElementById("semaine").value
const mois = document.getElementById("mois").value
const annee = document.getElementById("annee").value
const evaluateur = document.getElementById("evaluateur").value

const data = [

{
Jour:jour,
Semaine:semaine,
Mois:mois,
Annee:annee,
Evaluateur:evaluateur,
Nombre_de_criteres:nbCritere,
Score:scoreFinal
}

]

const ws = XLSX.utils.json_to_sheet(data)

const wb = XLSX.utils.book_new()

XLSX.utils.book_append_sheet(wb,ws,"Résumé contrôle")

XLSX.writeFile(wb,"resume_controle_audit.xlsx")

}
