<script>
    // Gestion dynamique des modes de contact
    const contactRadios = document.querySelectorAll('input[name="contact"]');
    const callbackDiv = document.getElementById('callbackOptions');
    const contactOptionsDivs = document.querySelectorAll('.contact-option');

    function updateContactUI() {
        const selectedRadio = document.querySelector('input[name="contact"]:checked');
        if (selectedRadio && selectedRadio.value === "callback") {
            callbackDiv.style.display = "block";
        } else {
            callbackDiv.style.display = "none";
        }
        // style visuel pour les cartes
        contactOptionsDivs.forEach(div => {
            const radio = div.querySelector('input');
            if (radio.checked) {
                div.classList.add('selected');
            } else {
                div.classList.remove('selected');
            }
        });
    }

    contactRadios.forEach(radio => {
        radio.addEventListener('change', updateContactUI);
    });
    updateContactUI();

    // Fonction toast
    function showToast(message, duration = 4000) {
        const toast = document.getElementById('toastNotif');
        document.getElementById('toastMessage').innerText = message;
        toast.classList.add('show');
        setTimeout(() => {
            toast.classList.remove('show');
        }, duration);
    }

    // Soumission formulaire + popup
    const form = document.getElementById('creditForm');
    const popup = document.getElementById('confirmationPopup');
    const dashboardSteps = document.getElementById('stepsContainer');

    form.addEventListener('submit', function(e) {
        e.preventDefault();
        
        // Vérification consentement
        const consent = document.getElementById('consentRGPD').checked;
        if (!consent) {
            alert("Merci d'accepter les conditions de traitement des données pour que nous puissions traiter votre demande.");
            return;
        }

        // Vérifier téléphone/email remplis
        const tel = document.getElementById('telephone').value.trim();
        const email = document.getElementById('email').value.trim();
        if (!tel || !email) {
            alert("Veuillez renseigner un téléphone et un email valides.");
            return;
        }

        // Récupération du mode de contact choisi
        const contactMode = document.querySelector('input[name="contact"]:checked');
        if (!contactMode) {
            alert("Veuillez sélectionner un mode de contact (rappel ou 100% en ligne).");
            return;
        }

        // simulation d'envoi
        const prenom = document.getElementById('prenom').value;
        const nom = document.getElementById('nom').value;
        let creneauMsg = "";
        if (contactMode.value === "callback") {
            const creneau = document.getElementById('creneauSelect').value;
            creneauMsg = ` - Créneau choisi : ${creneau}`;
            showToast(`✅ Demande enregistrée. Rappel planifié ${creneau}`, 5000);
        } else {
            showToast(`✅ Demande 100% en ligne reçue. Accédez à votre espace simulation.`, 5000);
        }
        
        // Sauvegarde localStorage pour dashboard
        const leadData = {
            prenom: prenom,
            nom: nom,
            tel: tel,
            email: email,
            contactMode: contactMode.value,
            creneau: contactMode.value === "callback" ? document.getElementById('creneauSelect').value : null,
            dateDemande: new Date().toLocaleString(),
            etape: 2  // analyse en cours
        };
