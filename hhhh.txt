Private Sub CommandButton2_Click()

    Dim wsDest As Worksheet
    Dim wsSource As Worksheet
    Dim wbSource As Workbook
    
    Dim lastRow As Long, lastCol As Long
    Dim formatCol As Long
    Dim i As Long, destRow As Long
    
    Dim fso As Object
    Dim folder As Object
    Dim subfolder As Object
    Dim fichier As Object
    
    Dim DossierChemin As String
    Dim sheetNames As Variant
    Dim shName As Variant
    
    ' === CHEMIN DOSSIER ===
    DossierChemin = "\\uf11pu01\ACCESSIBILITÉ_CONSEILLERS\Cofidis"
    
    ' === DESTINATION ===
    Set wsDest = ThisWorkbook.Worksheets(mois.Value)
    wsDest.Cells.Clear
    destRow = 1
    
    ' === FILE SYSTEM OBJECT ===
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set folder = fso.GetFolder(DossierChemin)
    
    ' === LISTE DES FEUILLES A TRAITER ===
    sheetNames = Array("OCAOCR", "ETUDEINC", "SOLVA")
    
    ' === BOUCLE SUR LES SOUS-DOSSIERS ===
    For Each subfolder In folder.SubFolders
        
        ' === BOUCLE SUR LES FICHIERS ===
        For Each fichier In subfolder.Files
            
            If LCase(fso.GetExtensionName(fichier.Name)) Like "xls*" Then
                
                Set wbSource = Workbooks.Open(fichier.Path, ReadOnly:=True)
                
                ' === BOUCLE SUR LES 3 FEUILLES ===
                For Each shName In sheetNames
                    
                    On Error Resume Next
                    Set wsSource = wbSource.Worksheets(shName)
                    On Error GoTo 0
                    
                    If Not wsSource Is Nothing Then
                        
                        lastCol = wsSource.Cells(1, wsSource.Columns.Count).End(xlToLeft).Column
                        lastRow = wsSource.Cells(wsSource.Rows.Count, 1).End(xlUp).Row
                        
                        ' === TROUVER COLONNE "fichier" ===
                        formatCol = 0
                        For i = 1 To lastCol
                            If LCase(wsSource.Cells(1, i).Value) = "fichier" Then
                                formatCol = i
                                Exit For
                            End If
                        Next i
                        
                        If formatCol > 0 Then
                            
                            ' === COPIER EN-TETE UNE SEULE FOIS ===
                            If destRow = 1 Then
                                wsSource.Rows(1).Copy wsDest.Rows(destRow)
                                destRow = destRow + 1
                            End If
                            
                            ' === COPIER LIGNES CORRESPONDANTES ===
                            For i = 2 To lastRow
                                
                                If InStr(1, LCase(wsSource.Cells(i, formatCol).Value), "i") > 0 Then
                                    
                                    wsSource.Rows(i).Copy wsDest.Rows(destRow)
                                    destRow = destRow + 1
                                    
                                End If
                                
                            Next i
                            
                        End If
                        
                    End If
                    
                    Set wsSource = Nothing
                    
                Next shName
                
                wbSource.Close False
                
            End If
            
        Next fichier
        
    Next subfolder
    
    MsgBox "Extraction terminée", vbInformation
    
End Sub
