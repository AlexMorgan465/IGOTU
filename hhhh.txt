function calculer() {
  const prix = parseFloat(document.getElementById('prix').value);
  const apport = parseFloat(document.getElementById('apport').value);
  const revenu = parseFloat(document.getElementById('revenu').value);
  const charges = parseFloat(document.getElementById('charges').value);
  const duree = parseInt(document.getElementById('duree').value);

  const montant = prix - apport;
  const taux = 0.02 / 12;
  const n = duree * 12;

  const mensualite = (montant * taux) / (1 - Math.pow(1 + taux, -n));

  document.getElementById('montant').innerText = montant.toLocaleString() + ' €';
  document.getElementById('mensualite').innerText = Math.round(mensualite) + ' € / mois';
}
