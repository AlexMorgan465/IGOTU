Private Sub btnDeplacer_Click()

    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    
    Dim cheminSource As String
    Dim cheminDestination As String
    
    ' Sélection du fichier source
    With Application.FileDialog(msoFileDialogFilePicker)
        .Title = "Sélectionner le fichier à déplacer"
        If .Show <> -1 Then Exit Sub
        cheminSource = .SelectedItems(1)
    End With
    
    ' Sélection du dossier destination
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = "Sélectionner le dossier destination"
        If .Show <> -1 Then Exit Sub
        cheminDestination = .SelectedItems(1)
    End With
    
    ' Vérification et déplacement
    If fso.FileExists(cheminSource) Then
        
        fso.MoveFile cheminSource, cheminDestination & "\" & _
                     Mid(cheminSource, InStrRev(cheminSource, "\") + 1)
        
        MsgBox "Fichier déplacé avec succès !", vbInformation
        
    Else
        MsgBox "Fichier introuvable !", vbExclamation
    End If

End Sub
