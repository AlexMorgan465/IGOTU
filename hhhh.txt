Private Sub cmdSave_Click()

    On Error GoTo ErrHandler
    
    ' Vérifie si le classeur a déjà été enregistré
    If ThisWorkbook.Path = "" Then
        MsgBox "Le classeur n'a pas encore été enregistré. Utilisez 'Enregistrer sous'.", vbExclamation
        Exit Sub
    End If
    
    ' Sauvegarde
    ThisWorkbook.Save
    
    MsgBox "Classeur sauvegardé avec succès.", vbInformation
    
    Exit Sub

ErrHandler:
    MsgBox "Erreur lors de la sauvegarde : " & Err.Description, vbCritical

End Sub
