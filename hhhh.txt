        // Fonction pour calculer le numéro de semaine ISO
        function getISOWeekNumber(date) {
            if (!date) return '';
            
            const d = new Date(date);
            d.setHours(0, 0, 0, 0);
            
            // Jeudi de la même semaine
            d.setDate(d.getDate() + 4 - (d.getDay() || 7));
            
            // Premier janvier de l'année
            const yearStart = new Date(d.getFullYear(), 0, 1);
            
            // Calcul du nombre de jours
            const weekNum = Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
            
            return weekNum;
        }

        // Mettre à jour le numéro de semaine
        function updateWeekNumber() {
            const dateInput = document.getElementById('doc-date');
            const weekInput = document.getElementById('week-number');
            
            if (dateInput.value) {
                const weekNum = getISOWeekNumber(dateInput.value);
                weekInput.value = `Semaine ${weekNum}`;
            } else {
                weekInput.value = '';
            }
        }

        // Calculer la note de qualité
        function calculateQualityNote() {
            const counts = {
                c: 0,
                nc: 0,
                na: 0,
                total: 0
            };
            
            Object.values(complianceData).forEach(item => {
                counts.total++;
                if (item.status === 'c') counts.c++;
                else if (item.status === 'nc') counts.nc++;
                else if (item.status === 'na') counts.na++;
            });
            
            const evaluatedCount = counts.c + counts.nc;
            if (evaluatedCount === 0) return 0;
            
            // Note sur 20 basée sur le pourcentage de conformité
            const conformitePourcentage = (counts.c / evaluatedCount) * 100;
            const note = (conformitePourcentage / 100) * 20;
            
            return note.toFixed(1);
        }

        // Configurer les écouteurs d'événements
        function setupEventListeners() {
            // Écouter les changements de statut de conformité
            document.addEventListener('change', function(e) {
                if (e.target.classList.contains('compliance-status')) {
                    const row = e.target.closest('tr');
                    const critere = row.getAttribute('data-critere');
                    const status = e.target.value;
                    const commentInput = row.querySelector('.comment-input');
                    
                    // Mettre à jour les données
                    complianceData[critere].status = status;
                    
                    // Gérer l'affichage du commentaire
                    if (status === 'nc') {
                        commentInput.classList.add('show');
                        commentInput.required = true;
                    } else {
                        commentInput.classList.remove('show');
                        commentInput.required = false;
                        complianceData[critere].comment = '';
                        commentInput.value = '';
                    }
                    
                    // Mettre à jour les statistiques
                    updateStatusSummary();
                    
                    // Mettre à jour la note de qualité
                    updateQualityNote();
                    
                    // Sauvegarder automatiquement
                    autoSave();
                }
            });
            
            // Écouter les changements de commentaires
            document.addEventListener('input', function(e) {
                if (e.target.classList.contains('comment-input')) {
                    const row = e.target.closest('tr');
                    const critere = row.getAttribute('data-critere');
                    complianceData[critere].comment = e.target.value;
                    
                    // Sauvegarder automatiquement
                    autoSave();
                }
            });
            
            // Mettre à jour le numéro de semaine quand la date change
            document.getElementById('doc-date').addEventListener('change', updateWeekNumber);
            
            // Bouton d'exportation Excel
            document.getElementById('export-btn').addEventListener('click', exportToExcel);
            
            // Bouton de réinitialisation
            document.getElementById('reset-btn').addEventListener('click', resetForm);
            
            // Bouton de sauvegarde
            document.getElementById('save-btn').addEventListener('click', saveData);
            
            // Écouter les changements dans les informations du document
            const docInputs = ['doc-name', 'doc-type', 'doc-date', 'evaluator', 'evalue'];
            docInputs.forEach(id => {
                document.getElementById(id).addEventListener('input', autoSave);
            });
        }

        // Mettre à jour le résumé des statuts
        function updateStatusSummary() {
            const counts = {
                c: 0,
                nc: 0,
                na: 0,
                'non-renseigne': 0
            };
            
            // Compter les statuts
            Object.values(complianceData).forEach(item => {

                    counts[item.status] = (counts[item.status] || 0) + 1;

                <div class="status-card">

                <div class="status-card">
                    <h3>Non conforme</h3>
                    <div class="status-value non-conforme">${counts.nc}</div>

                </div>
            `;
            
            document.getElementById('status-summary').innerHTML = summaryHTML;
        }

        // Fonction pour créer un fichier Excel bien formaté
        function exportToExcel() {
            // Vérifier que le nom du document est rempli
            const docName = document.getElementById('doc-name').value;
            if (!docName) {
                alert('Veuillez saisir le nom du document avant d\'exporter.');
                document.getElementById('doc-name').focus();
                return;
            }
            
            // Vérifier les commentaires pour les critères non conformes
            let missingComments = false;
            let missingCommentCritere = '';
            
            for (const [critere, data] of Object.entries(complianceData)) {
                if (data.status === 'nc' && !data.comment.trim()) {
                    missingComments = true;
                    missingCommentCritere = critere;
                    break;
                }
            }
            
            if (missingComments) {
                alert(`Le critère ${missingCommentCritere} est marqué comme "Non conforme" mais ne comporte pas de commentaire de justification. Veuillez ajouter un commentaire avant d'exporter.`);
                const row = document.querySelector(`[data-critere="${missingCommentCritere}"]`);
                if (row) {
                    row.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    const commentInput = row.querySelector('.comment-input');
                    commentInput.classList.add('show');
                    commentInput.focus();
                }
                return;
            }
            
            // Afficher l'indicateur de chargement
            document.getElementById('loading').style.display = 'block';
            
            // Créer un nouveau classeur Excel
            const wb = XLSX.utils.book_new();
            wb.Props = {
                Title: "Rapport de Conformité Accessibilité",
                Subject: "Audit Accessibilité",
                Author: "Formulaire Accessibilité",
                CreatedDate: new Date()
            };
            
            // Créer la feuille de données
            const ws_data = [];
            
            // 1. Ajouter les informations du document (uniquement dans la feuille Résumé)
            // Pas d'ajout ici car elles seront dans la feuille Résumé
            
            // 2. Ajouter l'en-tête du tableau
            ws_data.push(["Thème", "Critère", "Intitulé", "Conformité", "Commentaire"]);
            
            // 3. Ajouter les données des critères
            let currentTheme = "";
            let themeStartRow = 2; // Ligne de début du premier thème (après les en-têtes)
            let currentThemeRow = themeStartRow;
            let mergeRanges = []; // Pour stocker les plages de fusion
            
            criteriaData.forEach((item, index) => {
                const data = complianceData[item.critere];
                
                // Déterminer le texte de conformité
                let conformite = "";
                switch(data.status) {
                    case 'c': conformite = "Conforme"; break;
                    case 'nc': conformite = "Non conforme"; break;
                    case 'na': conformite = "Non applicable"; break;
                    default: conformite = "Non renseigné";
                }
                
                // Si c'est un nouveau thème
                if (item.theme && item.theme !== currentTheme) {
                    currentTheme = item.theme;
                    currentThemeRow = ws_data.length + 1; // +1 car ws_data est 0-indexed mais Excel est 1-indexed
                    
                    // Ajouter le thème
                    ws_data.push([currentTheme, item.critere, item.intitule, conformite, data.comment || ""]);
                } else {
                    // Sinon, ligne normale de critère
                    ws_data.push(["", item.critere, item.intitule, conformite, data.comment || ""]);
                }
                
                // Identifier la fin du thème pour les fusions
                if (index === criteriaData.length - 1 || 
                    (criteriaData[index + 1] && criteriaData[index + 1].theme)) {
                    // C'est la dernière ligne du thème
                    if (currentTheme !== "") {
                        const themeEndRow = ws_data.length; // Dernière ligne du thème
                        if (themeEndRow > currentThemeRow) {
                            // Ajouter la plage de fusion pour ce thème (colonnes A)
                            mergeRanges.push({
                                s: {r: currentThemeRow - 1, c: 0}, // Début (0-indexed)
                                e: {r: themeEndRow - 1, c: 0}      // Fin (0-indexed)
                            });
                        }
                    }
                }
            });
            
            // Convertir les données en feuille de calcul
            const ws = XLSX.utils.aoa_to_sheet(ws_data);
            
            // 4. Définir les largeurs de colonnes
            const colWidths = [
                { wch: 30 }, // Thème
                { wch: 10 }, // Critère
                { wch: 80 }, // Intitulé
                { wch: 15 }, // Conformité
                { wch: 50 }  // Commentaire
            ];
            ws['!cols'] = colWidths;
            
            // 5. Définir les plages de fusion
            ws['!merges'] = mergeRanges;
            
            // 6. Appliquer le style aux cellules
            const range = XLSX.utils.decode_range(ws['!ref']);
            
            // Parcourir toutes les cellules
            for (let R = range.s.r; R <= range.e.r; ++R) {
                for (let C = range.s.c; C <= range.e.c; ++C) {
                    const cell_address = {c: C, r: R};
                    const cell_ref = XLSX.utils.encode_cell(cell_address);
                    
                    if (!ws[cell_ref]) continue;
                    
                    let cellStyle = {};
                    
                    // Style pour l'en-tête du tableau
                    if (R === 0) {
                        cellStyle = {
                            font: { bold: true, color: { rgb: "FFFFFF" } },
                            fill: { fgColor: { rgb: "2C3E50" } },
                            alignment: { horizontal: "center", vertical: "center" },
                            border: {
                                top: { style: "thin", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            }
                        };
                    }
                    
                    // Style pour les thèmes
                    else if (R >= 1 && ws[cell_ref] && ws[cell_ref].v && ws[cell_ref].v.toString().match(/^\d+\.\s/)) {
                        cellStyle = {
                            font: { bold: true, sz: 12, color: { rgb: "000000" } },
                            alignment: { horizontal: "center", vertical: "center" },
                            border: {
                                top: { style: "medium", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            }
                        };
                        
                        // Appliquer la couleur du thème
                        const themeNum = parseInt(ws[cell_ref].v.split('.')[0]);
                        if (themeNum >= 1 && themeNum <= 10) {
                            cellStyle.fill = { fgColor: { rgb: themeColors[themeNum - 1] } };
                        }
                    }
                    
                    // Style pour les cellules de critères
                    else if (R > 0 && C >= 0 && C <= 4) {
                        cellStyle = {
                            border: {
                                top: { style: "thin", color: { rgb: "CCCCCC" } },
                                bottom: { style: "thin", color: { rgb: "CCCCCC" } },
                                left: { style: "thin", color: { rgb: "CCCCCC" } },
                                right: { style: "thin", color: { rgb: "CCCCCC" } }
                            },
                            alignment: { 
                                vertical: "top",
                                wrapText: true 
                            }
                        };
                        
                        // Appliquer la couleur de fond basée sur le thème
                        if (C === 0 && !ws[cell_ref].v) {
                            // Pour les cellules vides dans la colonne thème, on cherche le thème parent
                            let themeRow = R;
                            while (themeRow >= 1 && (!ws[XLSX.utils.encode_cell({c:0, r:themeRow})] || 
                                   !ws[XLSX.utils.encode_cell({c:0, r:themeRow})].v)) {
                                themeRow--;
                            }
                            if (themeRow >= 1 && ws[XLSX.utils.encode_cell({c:0, r:themeRow})]) {
                                const themeValue = ws[XLSX.utils.encode_cell({c:0, r:themeRow})].v;
                                const themeNum = parseInt(themeValue.split('.')[0]);
                                if (themeNum >= 1 && themeNum <= 10) {
                                    cellStyle.fill = { fgColor: { rgb: themeColors[themeNum - 1] + "80" } }; // 80 = 50% d'opacité
                                }
                            }
                        }
                        
                        // Style pour la colonne de conformité
                        if (C === 3) {
                            cellStyle.alignment = { horizontal: "center", vertical: "center" };
                            if (ws[cell_ref].v === "Conforme") {
                                cellStyle.font = { bold: true, color: { rgb: "006400" } }; // Vert foncé
                            } else if (ws[cell_ref].v === "Non conforme") {
                                cellStyle.font = { bold: true, color: { rgb: "8B0000" } }; // Rouge foncé
                            } else if (ws[cell_ref].v === "Non applicable") {
                                cellStyle.font = { color: { rgb: "696969" } }; // Gris
                            }
                        }
                    }
                    
                    // Appliquer le style
                    ws[cell_ref].s = cellStyle;
                }
            }
            
            // 7. Ajouter la feuille au classeur
            XLSX.utils.book_append_sheet(wb, ws, "Conformité");
            
            // 8. Créer une feuille de résumé avec les informations du document
            createSummarySheet(wb);
            
            // 9. Générer le fichier Excel
            setTimeout(() => {
                const wbout = XLSX.write(wb, {bookType:'xlsx', type:'binary'});
                
                function s2ab(s) {
                    const buf = new ArrayBuffer(s.length);
                    const view = new Uint8Array(buf);
                    for (let i=0; i<s.length; i++) view[i] = s.charCodeAt(i) & 0xFF;
                    return buf;
                }
                
                // Télécharger le fichier
                const blob = new Blob([s2ab(wbout)], {type:"application/octet-stream"});
                const link = document.createElement("a");
                link.href = URL.createObjectURL(blob);
                
                const safeFileName = docName.replace(/[^a-z0-9]/gi, '_').substring(0, 50);
                link.download = `RAPDF_${safeFileName}_${new Date().toISOString().slice(0,10)}.xlsx`;
                
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                
                // Masquer l'indicateur de chargement
                document.getElementById('loading').style.display = 'none';
                
                // Confirmation
                const nonConformeCount = Object.keys(complianceData).filter(c => complianceData[c].status === 'nc').length;
                const note = calculateQualityNote();
                alert(`Fichier Excel généré avec succès!\n\n${nonConformeCount} critère(s) sont marqués comme non conforme(s).\nNote de qualité: ${note}/20\n\nLe fichier inclut:\n✓ Les thèmes fusionnés\n✓ Les couleurs par thème\n✓ Le formatage professionnel\n✓ Une feuille de résumé avec toutes les informations`);
            }, 500);
        }

        // Créer une feuille de résumé
        function createSummarySheet(wb) {
            const summaryData = [];
            
            // Titre
            summaryData.push(["RAPPORT DE CONFORMITÉ ACCESSIBILITÉ"]);
            summaryData.push([]);
            
            // Informations du document
            summaryData.push(["INFORMATIONS DU DOCUMENT"]);
            summaryData.push(["Nom du document:", document.getElementById('doc-name').value || "Non spécifié"]);
            summaryData.push(["Type de document:", document.getElementById('doc-type').value || "Non spécifié"]);
            summaryData.push(["Date d'évaluation:", document.getElementById('doc-date').value || "Non spécifiée"]);
            summaryData.push(["Évaluateur:", document.getElementById('evaluator').value || "Non spécifié"]);
            summaryData.push(["Évalué (Agent):", document.getElementById('evalue').value || "Non spécifié"]);
            summaryData.push(["Numéro de semaine ISO:", document.getElementById('week-number').value || "Non calculé"]);
            summaryData.push(["Date d'export:", new Date().toLocaleDateString('fr-FR')]);
            summaryData.push([]);
            
            // Calculer les statistiques
            const counts = { c: 0, nc: 0, na: 0, total: 0 };
            Object.values(complianceData).forEach(item => {
                counts.total++;
                if (item.status === 'c') counts.c++;
                else if (item.status === 'nc') counts.nc++;
                else if (item.status === 'na') counts.na++;
            });
            
            const evaluatedCount = counts.c + counts.nc;
            const conformitePourcentage = evaluatedCount > 0 ? (counts.c / evaluatedCount * 100).toFixed(1) : 0;
            const note = (conformitePourcentage / 100 * 20).toFixed(1);
            
            // Note de qualité
            summaryData.push(["NOTE DE QUALITÉ"]);
            summaryData.push(["Note sur 20:", `${note}/20`]);
            summaryData.push(["Pourcentage de conformité:", `${conformitePourcentage}%`]);
            summaryData.push([]);
            
            // Tableau de résumé
            summaryData.push(["RÉSUMÉ DES RÉSULTATS"]);
            summaryData.push(["Statut", "Nombre", "Pourcentage"]);
            summaryData.push(["Conforme", counts.c, counts.total > 0 ? (counts.c / counts.total * 100).toFixed(1) + "%" : "0%"]);
            summaryData.push(["Non conforme", counts.nc, counts.total > 0 ? (counts.nc / counts.total * 100).toFixed(1) + "%" : "0%"]);
            summaryData.push(["Non applicable", counts.na, counts.total > 0 ? (counts.na / counts.total * 100).toFixed(1) + "%" : "0%"]);
            summaryData.push(["Non renseigné", counts.total - (counts.c + counts.nc + counts.na), 
                            counts.total > 0 ? ((counts.total - (counts.c + counts.nc + counts.na)) / counts.total * 100).toFixed(1) + "%" : "0%"]);
            summaryData.push(["TOTAL", counts.total, "100%"]);
            summaryData.push([]);
            
            // Détails par thème
            summaryData.push(["DÉTAILS PAR THÈME"]);
            summaryData.push(["Thème", "Conforme", "Non conforme", "Non applicable", "Non renseigné", "Total"]);
            
            const themeStats = {};
            criteriaData.forEach(item => {
                const themeNum = item.themeNumber;
                if (!themeStats[themeNum]) {
                    themeStats[themeNum] = { theme: item.theme || criteriaData.find(c => c.themeNumber === themeNum && c.theme).theme, c: 0, nc: 0, na: 0, nr: 0, total: 0 };
                }
                themeStats[themeNum].total++;
                const status = complianceData[item.critere].status;
                if (status === 'c') themeStats[themeNum].c++;
                else if (status === 'nc') themeStats[themeNum].nc++;
                else if (status === 'na') themeStats[themeNum].na++;
                else themeStats[themeNum].nr++;
            });
            
            // Ajouter les statistiques par thème
            Object.values(themeStats).forEach(stat => {
                if (stat.theme) {
                    summaryData.push([stat.theme, stat.c, stat.nc, stat.na, stat.nr, stat.total]);
                }
            });
            
            // Créer la feuille de résumé
            const ws_summary = XLSX.utils.aoa_to_sheet(summaryData);
            
            // Largeurs de colonnes
            ws_summary['!cols'] = [
                { wch: 40 },
                { wch: 20 },
                { wch: 15 },
                { wch: 15 },
                { wch: 15 },
                { wch: 15 }
            ];
            
            // Appliquer des styles
            const range = XLSX.utils.decode_range(ws_summary['!ref']);
            for (let R = range.s.r; R <= range.e.r; ++R) {
                for (let C = range.s.c; C <= range.e.c; ++C) {
                    const cell_address = {c: C, r: R};
                    const cell_ref = XLSX.utils.encode_cell(cell_address);
                    
                    if (!ws_summary[cell_ref]) continue;
                    
                    let cellStyle = {};
                    
                    // Titre principal
                    if (R === 0) {
                        cellStyle = {
                            font: { sz: 18, bold: true, color: { rgb: "2C3E50" } },
                            alignment: { horizontal: "center" }
                        };
                    }
                    
                    // Titres de sections
                    else if (R === 2 || R === 10 || R === 16) {
                        cellStyle = {
                            font: { sz: 14, bold: true, color: { rgb: "FFFFFF" } },
                            fill: { fgColor: { rgb: "4A6491" } }
                        };
                    }
                    
                    // En-têtes des tableaux
                    else if (R === 3 || R === 11 || R === 17) {
                        cellStyle = {
                            font: { bold: true, color: { rgb: "FFFFFF" } },
                            fill: { fgColor: { rgb: "2C3E50" } },
                            border: {
                                top: { style: "thin", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            }
                        };
                    }
                    
                    // Données
                    else if (R > 3 && R <= 8 || R > 11 && R <= 14 || R > 17) {
                        cellStyle = {
                            border: {
                                bottom: { style: "thin", color: { rgb: "CCCCCC" } },
                                left: { style: "thin", color: { rgb: "CCCCCC" } },
                                right: { style: "thin", color: { rgb: "CCCCCC" } }
                            }
                        };
                        
                        // Mettre en évidence la ligne de note
                        if (R === 12) {
                            cellStyle.font = { bold: true, color: { rgb: "006400" } };
                        }
                    }
                    
                    ws_summary[cell_ref].s = cellStyle;
                }
            }
            
            // Ajouter la feuille au classeur
            XLSX.utils.book_append_sheet(wb, ws_summary, "Résumé");
        }

        // Réinitialiser le formulaire
        function resetForm() {
            if (confirm("Êtes-vous sûr de vouloir réinitialiser tout le formulaire ? Toutes les données seront perdues.")) {
                // Réinitialiser les informations du document
                document.getElementById('doc-name').value = '';
                document.getElementById('doc-type').value = '';
                document.getElementById('evaluator').value = '';
                document.getElementById('evalue').value = '';
                document.getElementById('doc-date').valueAsDate = new Date();
                updateWeekNumber();
                
                // Réinitialiser les données de conformité
                initializeComplianceData();
                
                // Réinitialiser l'interface
                document.querySelectorAll('.compliance-status').forEach(select => {
                    select.value = '';
                });
                
                document.querySelectorAll('.comment-input').forEach(input => {
                    input.value = '';
                    input.classList.remove('show');
                });
                
                // Mettre à jour les statistiques
                updateStatusSummary();
                
                // Mettre à jour la note de qualité
                updateQualityNote();
                
                // Supprimer les données sauvegardées
                localStorage.removeItem('accessibilityComplianceData');
                
                alert("Formulaire réinitialisé avec succès.");
            }
        }

        // Sauvegarder les données
        function saveData() {
            const docName = document.getElementById('doc-name').value;
            if (!docName) {
                alert('Veuillez saisir le nom du document avant de sauvegarder.');
                document.getElementById('doc-name').focus();
                return;
            }
            
            const saveData = {
                documentInfo: {
                    name: docName,
                    type: document.getElementById('doc-type').value,
                    date: document.getElementById('doc-date').value,
                    evaluator: document.getElementById('evaluator').value,
                    evalue: document.getElementById('evalue').value,
                    weekNumber: document.getElementById('week-number').value
                },
                complianceData: complianceData,
                timestamp: new Date().toISOString()
            };
            
            localStorage.setItem('accessibilityComplianceData', JSON.stringify(saveData));
            alert('Données sauvegardées avec succès. Vous pouvez recharger cette page à tout moment pour continuer.');
        }

        // Charger les données sauvegardées
        function loadSavedData() {
            const savedData = localStorage.getItem('accessibilityComplianceData');
            if (savedData) {
                try {
                    const data = JSON.parse(savedData);
                    
                    // Charger les informations du document
                    if (data.documentInfo) {
                        document.getElementById('doc-name').value = data.documentInfo.name || '';
                        document.getElementById('doc-type').value = data.documentInfo.type || '';
                        document.getElementById('doc-date').value = data.documentInfo.date || '';
                        document.getElementById('evaluator').value = data.documentInfo.evaluator || '';
                        document.getElementById('evalue').value = data.documentInfo.evalue || '';
                        
                        // Mettre à jour le numéro de semaine
                        updateWeekNumber();
                    }
                    
                    // Charger les données de conformité
                    if (data.complianceData) {
                        Object.assign(complianceData, data.complianceData);
                        
                        // Mettre à jour l'interface
                        Object.entries(complianceData).forEach(([critere, itemData]) => {
                            const row = document.querySelector(`[data-critere="${critere}"]`);
                            if (row) {
                                const statusSelect = row.querySelector('.compliance-status');
                                const commentInput = row.querySelector('.comment-input');
                                
                                if (statusSelect) {
                                    statusSelect.value = itemData.status || '';
                                    
                                    if (itemData.status === 'nc') {
                                        commentInput.classList.add('show');
                                    }
                                }
                                
                                if (commentInput && itemData.comment) {
                                    commentInput.value = itemData.comment;
                                }
                            }
                        });
                        
                        // Mettre à jour les statistiques
                        updateStatusSummary();
                        
                        // Mettre à jour la note de qualité
                        updateQualityNote();
                    }
                } catch (e) {
                    console.error('Erreur lors du chargement des données sauvegardées:', e);
                }
            }
        }
