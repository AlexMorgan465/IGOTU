<script>

let resumeData = {}

function analyser(){

const file = document.getElementById("excelFile").files[0]

const reader = new FileReader()

reader.onload = function(e){

const data = new Uint8Array(e.target.result)

const workbook = XLSX.read(data,{type:'array'})

const sheet = workbook.Sheets[workbook.SheetNames[0]]

const json = XLSX.utils.sheet_to_json(sheet)

let C = 0
let NC = 0
let bloquantNC = 0

json.forEach(row=>{

let status = (row["Status"] || "").toString().trim()
let bloquant = (row["Bloquant"] || "").toString().trim()

if(status === "C") C++

if(status === "NC"){
NC++

if(bloquant === "O"){
bloquantNC++
}
}

})

let total = C + NC

let score = 0

if(total > 0){
score = (C / total) * 100
}

score = score - (bloquantNC * 30)

if(score < 0) score = 0

score = Math.round(score)

document.getElementById("c").innerText = C
document.getElementById("nc").innerText = NC
document.getElementById("bloquant").innerText = bloquantNC
document.getElementById("score").innerText = score

resumeData = {
C,
NC,
bloquantNC,
score
}

}

reader.readAsArrayBuffer(file)

}

function exporter(){

const jour = document.getElementById("jour").value
const semaine = document.getElementById("semaine").value
const mois = document.getElementById("mois").value
const annee = document.getElementById("annee").value
const manager = document.getElementById("manager").value

let data = [
{
Jour:jour,
Semaine:semaine,
Mois:mois,
Annee:annee,
Manager:manager,
Critere_C:resumeData.C,
Critere_NC:resumeData.NC,
Bloquant_NC:resumeData.bloquantNC,
Score:resumeData.score
}
]

const ws = XLSX.utils.json_to_sheet(data)

const wb = XLSX.utils.book_new()

XLSX.utils.book_append_sheet(wb,ws,"Resume")

XLSX.writeFile(wb,"resume_controle.xlsx")

}
