Private Sub CommandButton_CreateFolder_Click()
    Dim fso As Object
    Dim cheminParent As String
    Dim nomDossier As String
    Dim nouveauChemin As String
    
    ' Vérifier qu’un noeud est sélectionné
    If TreeView1.SelectedItem Is Nothing Then
        MsgBox "Sélectionne un dossier d'abord", vbExclamation
        Exit Sub
    End If
    
    cheminParent = TreeView1.SelectedItem.Key
    
    ' Demander le nom du dossier
    nomDossier = InputBox("Nom du nouveau dossier :", "Créer dossier")
    
    If nomDossier = "" Then Exit Sub
    
    nouveauChemin = cheminParent & "\" & nomDossier
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    
    ' Vérifier si existe déjà
    If fso.FolderExists(nouveauChemin) Then
        MsgBox "Ce dossier existe déjà", vbExclamation
        Exit Sub
    End If
    
    ' Créer le dossier
    fso.CreateFolder nouveauChemin
    
    MsgBox "Dossier créé avec succès", vbInformation
    
    ' Rafraîchir TreeView
    TreeView1.Nodes.Clear
    Call ChargerDossier("C:\Test", Nothing)
End Sub
Private Sub CommandButton_OpenFile_Click()
    Dim cheminDossier As String
    Dim fichier As String
    Dim cheminComplet As String
    
    ' Vérifier sélection
    If TreeView1.SelectedItem Is Nothing Then
        MsgBox "Sélectionne un dossier", vbExclamation
        Exit Sub
    End If
    
    If ListBox1.ListIndex = -1 Then
        MsgBox "Sélectionne un fichier", vbExclamation
        Exit Sub
    End If
    
    fichier = ListBox1.Value
    
    ' Ignorer les dossiers
    If Left(fichier, 9) = "[Dossier]" Then
        MsgBox "Ceci est un dossier", vbExclamation
        Exit Sub
    End If
    
    cheminDossier = TreeView1.SelectedItem.Key
    cheminComplet = cheminDossier & "\" & fichier
    
    ' Ouvrir le fichier avec l'application par défaut
    Shell "explorer.exe """ & cheminComplet & """", vbNormalFocus
End Sub
