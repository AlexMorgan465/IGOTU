Private Sub ParcourirDossier(ByVal dossier As Object)

    Dim sousDossier As Object
    Dim categorie As Object
    Dim dossierFinal As Object
    
    For Each sousDossier In dossier.SubFolders
        
        If UCase(sousDossier.Name) = "NON CONFORME" Then
            
            ' CAS SPECIAL ACM
            If UCase(racineSelectionnee) = "ACM" Then
                
                For Each categorie In sousDossier.SubFolders
                    For Each dossierFinal In categorie.SubFolders
                        
                        lstResultats.AddItem dossierFinal.Name
                        lstResultats.List(lstResultats.ListCount - 1, 1) = dossierFinal.Path
                        
                    Next dossierFinal
                Next categorie
            
            ' TOUS LES AUTRES DOSSIERS
            Else
                
                For Each dossierFinal In sousDossier.SubFolders
                    
                    lstResultats.AddItem dossierFinal.Name
                    lstResultats.List(lstResultats.ListCount - 1, 1) = dossierFinal.Path
                    
                Next dossierFinal
                
            End If
            
        End If
        
        Call ParcourirDossier(sousDossier)
        
    Next sousDossier

End Sub
