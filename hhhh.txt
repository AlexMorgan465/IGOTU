    // SIMULATION COMPLÈTE – FONCTIONNE SANS ALTERER LE STYLE EXISTANT

    (function() {
        // DOM elements
        const prixInput = document.getElementById('input-prix');
        const apportRange = document.getElementById('input-apport');
        const apportSpan = document.getElementById('val-apport');
        const revenusInput = document.getElementById('input-revenus');
        const chargesInput = document.getElementById('input-charges');
        const jobSelect = document.getElementById('input-job');
        const resMontantSpan = document.getElementById('res-montant');
        const resMensualiteSpan = document.getElementById('res-mensualite');
        const resApportSpan = document.getElementById('res-apport');
        const resTauxSpan = document.getElementById('res-taux');

        let bienType = 'Ancien';   // par défaut

        // Gestion des toggles (type de bien)
        const toggleBtns = document.querySelectorAll('#bienToggle .btn-toggle');
        toggleBtns.forEach(btn => {
            btn.addEventListener('click', function() {
                toggleBtns.forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                bienType = this.getAttribute('data-type');
                calculer();
            });
        });

        // Mise à jour dynamique de l'affichage de l'apport (slider)
        function updateApportDisplay() {
            let val = parseInt(apportRange.value, 10);
            if (isNaN(val)) val = 0;
            apportSpan.innerText = val.toLocaleString('fr-FR');
            calculer();
        }

        apportRange.addEventListener('input', updateApportDisplay);
        
        // Écouteurs sur tous les champs impactant le calcul
        prixInput.addEventListener('input', function() { 
            let prix = parseFloat(prixInput.value);
            if (!isNaN(prix)) {
                // Ajuster max apport dynamiquement : ne peut pas dépasser le prix du bien
                let maxApport = Math.min(prix, 500000);
                apportRange.max = maxApport;
                let currentApport = parseFloat(apportRange.value);
                if (currentApport > prix) {
                    apportRange.value = prix;
                    updateApportDisplay();
                }
            }
            calculer();
        });
        revenusInput.addEventListener('input', calculer);
        chargesInput.addEventListener('input', calculer);
        jobSelect.addEventListener('change', calculer);
        
        // Fonction pour obtenir le taux annuel selon le bien et la profession
        function getTauxAnnuel() {
            let base = (bienType === 'Ancien') ? 0.0120 : 0.0100; // 1.20% ancien, 1.00% neuf
            let job = jobSelect.value;
            if (job === 'cdi') base -= 0.0005;   // -0.05%
            else if (job === 'independant') base += 0.0020; // +0.20%
            else if (job === 'cdd') base += 0.0010;  // +0.10%
            // Limites de sécurité
            base = Math.min(Math.max(base, 0.008), 0.018);
            return base;
        }

        // Calcul mensualité à partir du capital emprunté, taux annuel, durée 25 ans
        function calculerMensualite(capital, tauxAnnuel) {
            if (capital <= 0) return 0;
            const mois = 25 * 12;
            const tauxMensuel = tauxAnnuel / 12;
            if (tauxMensuel === 0) return capital / mois;
            const numerator = capital * tauxMensuel * Math.pow(1 + tauxMensuel, mois);
            const denominator = Math.pow(1 + tauxMensuel, mois) - 1;
            return numerator / denominator;
        }

        // Calcul du montant maximum finançable selon revenus / charges / taux
        function getMaxLoanFromCapacity(revenus, chargesMensuelles, tauxAnnuel) {
            const capaciteMensuelle = (revenus * 0.33) - chargesMensuelles;
            if (capaciteMensuelle <= 0) return 0;
            const mois = 25 * 12;
            const tauxMensuel = tauxAnnuel / 12;
            if (tauxMensuel === 0) return capaciteMensuelle * mois;
            // formule V = C * (1 - (1+r)^-n) / r
            const maxLoan = capaciteMensuelle * (1 - Math.pow(1 + tauxMensuel, -mois)) / tauxMensuel;
            return Math.floor(maxLoan);
        }

        // Fonction principale de calcul et mise à jour UI
        function calculer() {
            // Lecture des valeurs
            let prix = parseFloat(prixInput.value);
            let apport = parseFloat(apportRange.value);
            let revenus = parseFloat(revenusInput.value);
            let charges = parseFloat(chargesInput.value);
            
            if (isNaN(prix)) prix = 0;
            if (isNaN(apport)) apport = 0;
            if (isNaN(revenus)) revenus = 0;
            if (isNaN(charges)) charges = 0;
            
            // Apport max ne dépasse pas le prix
            if (apport > prix) apport = prix;
            const capitalNecessaire = Math.max(0, prix - apport);
            
            // Taux dynamique
            const tauxAnnuel = getTauxAnnuel();
            const tauxPercent = (tauxAnnuel * 100).toFixed(2).replace('.', ',');
            resTauxSpan.innerText = `${tauxPercent} %`;
            
            // Capacité d'emprunt maximale selon revenus / charges
            const maxLoanPossible = getMaxLoanFromCapacity(revenus, charges, tauxAnnuel);
            const montantFinancable = maxLoanPossible;
            resMontantSpan.innerText = montantFinancable.toLocaleString('fr-FR');
            
            // Mensualité basée sur le capital réellement nécessaire (prix - apport)
            let mensualiteReelle = 0;
            if (capitalNecessaire > 0) {
                mensualiteReelle = calculerMensualite(capitalNecessaire, tauxAnnuel);
            }
            resMensualiteSpan.innerText = Math.round(mensualiteReelle).toLocaleString('fr-FR');
            resApportSpan.innerText = apport.toLocaleString('fr-FR') + ' €';
            
            // Affichage optionnel d'un message d'alerte discrète si le projet dépasse la capacité (non intrusif)
            if (capitalNecessaire > montantFinancable && montantFinancable > 0) {
                // On ajoute une petite indication sans casser le style : juste une bordure légère en orange sur le montant finançable
                const montantElem = document.getElementById('res-montant');
                montantElem.style.color = "#c0392b";
                montantElem.title = "Votre projet dépasse la capacité d'emprunt estimée.";
            } else {
                const montantElem = document.getElementById('res-montant');
                montantElem.style.color = "#1a3a5f";
                montantElem.title = "";
            }
        }

        // --- STEPPER NAVIGATION (sans altérer style) ---
        let currentStep = 1;
        const steps = document.querySelectorAll('.form-step');
        const stepIndicators = document.querySelectorAll('.step');
        const prevBtn = document.getElementById('prevBtn1');
        const nextBtn = document.getElementById('nextBtn1');
        
        function updateStepVisibility() {
            steps.forEach((step, idx) => {
                if (idx + 1 === currentStep) {
                    step.classList.add('active');
                } else {
                    step.classList.remove('active');
                }
            });
            stepIndicators.forEach((ind, idx) => {
                if (idx + 1 === currentStep) {
                    ind.classList.add('active');
                } else {
                    ind.classList.remove('active');
                }
            });
            prevBtn.style.display = currentStep === 1 ? 'none' : 'block';
            nextBtn.innerText = currentStep === 5 ? "VOIR MON RÉSULTAT" : "ÉTAPE SUIVANTE";
        }
        
        window.changeStep = function(delta) {
            let newStep = currentStep + delta;
            if (newStep < 1) newStep = 1;
            if (newStep > 5) newStep = 5;
            currentStep = newStep;
            updateStepVisibility();
            // Option : quand on arrive à l'étape 5, on force le calcul (déjà fait)
            if (currentStep === 5) {
                calculer();
            }
        };
        
        nextBtn.addEventListener('click', () => {
            if (currentStep === 5) {
                // Dernière étape : scroll vers le résumé et met à jour
                document.querySelector('.summary-card').scrollIntoView({ behavior: 'smooth', block: 'center' });
                calculer();
            } else {
                changeStep(1);
            }
        });
        prevBtn.addEventListener('click', () => changeStep(-1));
        
        // Initialisation de tous les événements + premiers calculs
        function initSimulation() {
            updateApportDisplay();
            // Ajuster max apport en fonction du prix initial
            let initialPrix = parseFloat(prixInput.value);
            if (!isNaN(initialPrix)) {
                apportRange.max = Math.min(initialPrix, 500000);
                if (parseFloat(apportRange.value) > initialPrix) apportRange.value = initialPrix;
                updateApportDisplay();
            }
            calculer();
            updateStepVisibility();
            
            // Fixer le slider pour qu'il réponde au step2 si besoin
            // Ajout sécurité: si apport dépasse prix
            prixInput.dispatchEvent(new Event('input'));
        }
        
        // Bouton "OBTENIR UNE OFFRE !" – simulation de scroll vers formulaire contact
        const offreBtn = document.getElementById('btnObtenirOffre');
        if (offreBtn) {
            offreBtn.addEventListener('click', () => {
                const contactForm = document.querySelector('.form-card');
                if (contactForm) {
                    contactForm.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    // Petit feedback sympathique
                    alert("✅ Un conseiller peut vous recontacter pour finaliser votre offre personnalisée. Remplissez le formulaire ci-dessus.");
                } else {
                    alert("Merci pour votre intérêt ! Rapprochez-vous de votre conseiller Crédit Mutuel.");
                }
            });
        }
        
        // Lancement
        initSimulation();
        
        // Exposer globalement la fonction calculer pour d'éventuels appels externes
        window.calculer = calculer;
        window.updateVal = function(type, value) {
            if (type === 'bien') {
                bienType = value;
                toggleBtns.forEach(btn => {
                    if (btn.getAttribute('data-type') === value) {
                        btn.classList.add('active');
                    } else {
                        btn.classList.remove('active');
                    }
                });
                calculer();
            }
        };
    })();
