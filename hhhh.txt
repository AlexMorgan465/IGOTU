Private Sub CommandButton_CopyFile_Click()

    Dim fso As Object
    Dim cheminDossier As String
    Dim fichier As String
    Dim source As String
    Dim destination As String
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    
    ' Vérifier sélection fichier
    If TreeView1.SelectedItem Is Nothing Then
        MsgBox "Sélectionne un dossier", vbExclamation
        Exit Sub
    End If
    
    If ListBox1.ListIndex = -1 Then
        MsgBox "Sélectionne un fichier", vbExclamation
        Exit Sub
    End If
    
    fichier = ListBox1.Value
    
    ' Empêcher si c'est un dossier
    If Left(fichier, 9) = "[Dossier]" Then
        MsgBox "Veuillez sélectionner un fichier uniquement", vbExclamation
        Exit Sub
    End If
    
    cheminDossier = TreeView1.SelectedItem.Key
    source = cheminDossier & "\" & fichier
    
    ' Fenêtre parcourir
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = "Choisir le dossier de destination"
        If .Show <> -1 Then Exit Sub
        destination = .SelectedItems(1)
    End With
    
    On Error GoTo Erreur
    
    ' Copier fichier (overwrite activé)
    fso.CopyFile source, destination, True
    
    MsgBox "Fichier copié avec succès", vbInformation
    Exit Sub

Erreur:
    MsgBox "Erreur : " & Err.Description, vbCritical

End Sub
