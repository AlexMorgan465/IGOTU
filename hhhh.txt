Option Explicit

Private Sub UserForm_Initialize()
    Dim chemin As String
    
    ' Dossier racine (à modifier)
    chemin = "C:\Test"
    
    TreeView1.Nodes.Clear
    Call ChargerDossier(chemin, Nothing)
End Sub

' Fonction pour charger les dossiers dans le TreeView
Private Sub ChargerDossier(ByVal chemin As String, ByVal parentNode As Node)
    Dim fso As Object
    Dim dossier As Object
    Dim sousDossier As Object
    Dim node As Node
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set dossier = fso.GetFolder(chemin)
    
    ' Ajouter le noeud
    If parentNode Is Nothing Then
        Set node = TreeView1.Nodes.Add(, , chemin, dossier.Name)
    Else
        Set node = TreeView1.Nodes.Add(parentNode, tvwChild, chemin, dossier.Name)
    End If
    
    ' Parcourir les sous-dossiers
    For Each sousDossier In dossier.SubFolders
        Call ChargerDossier(sousDossier.Path, node)
    Next
End Sub

' Quand on clique sur un dossier
Private Sub TreeView1_NodeClick(ByVal Node As MSComctlLib.Node)
    Dim fso As Object
    Dim dossier As Object
    Dim fichier As Object
    
    ListBox1.Clear
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set dossier = fso.GetFolder(Node.Key)
    
    ' Afficher les fichiers
    For Each fichier In dossier.Files
        ListBox1.AddItem fichier.Name
    Next
    
    ' Afficher aussi les sous-dossiers (optionnel)
    Dim sousDossier As Object
    For Each sousDossier In dossier.SubFolders
        ListBox1.AddItem "[Dossier] " & sousDossier.Name
    Next
End Sub
