        // SIMULATEUR MINI
        function calculerMini() {
            const montant = parseFloat(document.getElementById('montantMini').value) || 200000;
            const apport = parseFloat(document.getElementById('apportMini').value) || 0;
            const duree = parseFloat(document.getElementById('dureeMini').value) || 20;
            const emprunt = montant - apport;
            const tauxMensuel = 0.035 / 12; // 3.5% annuel
            const mois = duree * 12;
            if (emprunt <=0 || mois<=0) { document.getElementById('resultMini').innerText = '0 €/mois'; return; }
            const mensualite = emprunt * tauxMensuel * Math.pow(1+tauxMensuel, mois) / (Math.pow(1+tauxMensuel, mois)-1);
            document.getElementById('resultMini').innerText = Math.round(mensualite) + ' €/mois';
        }
        window.onload = function() { calculerMini(); };

        // SIMULATION DÉTAILLÉE
        function calculerDetail() {
            const prix = parseFloat(document.getElementById('prix').value) || 250000;
            const apport = parseFloat(document.getElementById('apport').value) || 0;
            const duree = parseFloat(document.getElementById('duree').value) || 20;
            const revenus = parseFloat(document.getElementById('revenus').value) || 4000;
            const charges = parseFloat(document.getElementById('charges').value) || 0;
            const emprunt = prix - apport;
            const taux = 0.038/12; // 3.8% 
            const mois = duree * 12;
            let mensualite = 0;
            if (emprunt>0 && mois>0) {
                mensualite = emprunt * taux * Math.pow(1+taux, mois) / (Math.pow(1+taux, mois)-1);
            }
            const totalInterets = mensualite * mois - emprunt;
            const capacite = (revenus - charges) * 0.35; // règle 35% endettement
            document.getElementById('mensuRes').innerText = Math.round(mensualite) + ' €';
            document.getElementById('coutTotal').innerText = Math.round(totalInterets) + ' €';
            document.getElementById('capacite').innerText = Math.round(capacite) + ' €';
            document.getElementById('resultatDetail').style.display = 'flex';
        }

            document.querySelectorAll('.etape').forEach(e => e.classList.remove('active'));

            document.querySelectorAll('.creneau-btn').forEach(b => b.classList.remove('selected'));

            }
        }
