Private Sub ChargerArborescence()

    Dim chemin As String
    
    chemin = TextBox_Path.Value
    
    ' Vérifier si vide
    If chemin = "" Then
        MsgBox "Veuillez entrer un chemin", vbExclamation
        Exit Sub
    End If
    
    ' Vérifier si existe
    If Dir(chemin, vbDirectory) = "" Then
        MsgBox "Chemin invalide", vbCritical
        Exit Sub
    End If
    
    ' Vider
    TreeView1.Nodes.Clear
    ListBox1.Clear
    
    ' Charger
    Call ChargerDossier(chemin, Nothing)

End Sub
