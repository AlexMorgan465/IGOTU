Private Sub cmdLoad_Click()

Dim total As Integer
Dim ch As ChartObject

'KPI
total = Sheets("Sheet1").Range("B2") + Sheets("Sheet1").Range("B3")
lblTotal.Caption = "Total Sales : " & total

'Get chart
Set ch = Sheets("Sheet1").ChartObjects("ChartSales")

'Copy chart
ch.Chart.ChartArea.Copy

'Paste inside UserForm
Me.Paste

End Sub
