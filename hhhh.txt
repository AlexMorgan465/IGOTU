            function getISOWeekNumber(dateStr) {
                if (!dateStr) return '';
                const d = new Date(dateStr);
                d.setHours(0, 0, 0, 0);
                d.setDate(d.getDate() + 4 - (d.getDay() || 7));
                const yearStart = new Date(d.getFullYear(), 0, 1);
                return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
            }

            function updateWeekNumber() {
                const dateInput = document.getElementById('doc-date');
                const weekInput = document.getElementById('week-number');
                weekInput.value = dateInput.value ? `Semaine ${getISOWeekNumber(dateInput.value)}` : '';
            }

            function calculateStats() {
                const counts = { c: 0, nc: 0, na: 0, nr: 0 };
                Object.values(complianceData).forEach(item => {
                    if (item.status === 'c') counts.c++;
                    else if (item.status === 'nc') counts.nc++;
                    else if (item.status === 'na') counts.na++;
                    else counts.nr++;
                });
                counts.total = CRITERIA_DATA.length;
                return counts;
            }

            function calculateQualityNote() {
                const counts = calculateStats();
                const evaluated = counts.c + counts.nc;
                return evaluated > 0 ? (counts.c / evaluated) * 100 : 0;
            }

            function getNoteColor(note) {
                return note >= 85 ? "27ae60" : "e74c3c";
            }

                        tbody.appendChild(themeRow);
                    }

                    const row = document.createElement('tr');
                    row.dataset.critere = item.critere;
                    row.dataset.theme = item.themeNumber;
                    row.style.backgroundColor = `#${THEME_COLORS[item.themeNumber-1]}80`;

                    row.innerHTML = `
                        <td>${item.theme || ''}</td>
                        <td class="critere-num">${item.critere}</td>
                        <td>${item.intitule}</td>
                        <td class="compliance-select">
                            <select class="compliance-status">
                                <option value="">Sélectionner</option>
                                <option value="c">Conforme</option>
                                <option value="nc">Non conforme</option>
                                <option value="na">Non applicable</option>
                            </select>
                        </td>
                        <td class="comment-cell">
                            <textarea class="comment-input" placeholder="Justification obligatoire si non conforme..." rows="2"></textarea>
                        </td>
                    `;
                    tbody.appendChild(row);
                });
            }

            // ==================== CHARGEMENT/SAUVEGARDE ====================
            function loadFromStorage() {
                const saved = localStorage.getItem('accessibilityComplianceData');
                if (!saved) return;

                try {
                    const data = JSON.parse(saved);
                    if (data.documentInfo) {
                        document.getElementById('doc-name').value = data.documentInfo.name || '';
                        document.getElementById('doc-type').value = data.documentInfo.type || '';
                        document.getElementById('doc-date').value = data.documentInfo.date || '';
                        document.getElementById('evaluator').value = data.documentInfo.evaluator || '';
                        document.getElementById('evalue').value = data.documentInfo.evalue || '';
                        updateWeekNumber();
                    }

                    if (data.complianceData) {
                        complianceData = data.complianceData;
                        document.querySelectorAll('[data-critere]').forEach(row => {
                            const critere = row.dataset.critere;
                            const itemData = complianceData[critere];
                            if (!itemData) return;

                            const select = row.querySelector('.compliance-status');
                            const comment = row.querySelector('.comment-input');
                            
                            if (select) select.value = itemData.status || '';
                            if (comment) {
                                comment.value = itemData.comment || '';
                                if (itemData.status === 'nc') comment.classList.add('show');
                            }
                        });
                        updateStatusSummary();
                        updateQualityNoteDisplay();
                    }
                } catch (e) {
                    console.error('Erreur de chargement:', e);
                }
            }

            function saveToStorage() {
                const docName = document.getElementById('doc-name').value;
                if (!docName) {
                    alert('Veuillez saisir le nom du document');
                    document.getElementById('doc-name').focus();
                    return false;
                }

                const saveData = {
                    documentInfo: {
                        name: docName,
                        type: document.getElementById('doc-type').value,
                        date: document.getElementById('doc-date').value,
                        evaluator: document.getElementById('evaluator').value,
                        evalue: document.getElementById('evalue').value,
                        weekNumber: document.getElementById('week-number').value
                    },
                    complianceData: complianceData,
                    timestamp: new Date().toISOString()
                };
                
                localStorage.setItem('accessibilityComplianceData', JSON.stringify(saveData));
                return true;
            }

            // ==================== EXPORT EXCEL ====================
            function createSummarySheet(wb) {
                const docName = document.getElementById('doc-name').value || "Non spécifié";
                const docType = document.getElementById('doc-type').value || "Non spécifié";
                const docDate = document.getElementById('doc-date').value || "Non spécifiée";
                const evaluator = document.getElementById('evaluator').value || "Non spécifié";
                const evalue = document.getElementById('evalue').value || "Non spécifié";
                const weekNumber = document.getElementById('week-number').value || "Non calculé";

                const counts = calculateStats();
                const evaluated = counts.c + counts.nc;
                const note = evaluated > 0 ? (counts.c / evaluated) * 100 : 0;
                const status = note >= 85 ? "CONFORME (≥85%)" : "NON CONFORME (<85%)";

                // Construction des données
                const summaryData = [
                    ["RAPPORT DE CONFORMITÉ ACCESSIBILITÉ"],
                    [],
                    ["INFORMATIONS DU DOCUMENT"],
                    ["Nom du document:", docName],
                    ["Type de document:", docType],
                    ["Date d'évaluation:", docDate],
                    ["Évaluateur:", evaluator],
                    ["Évalué (Agent):", evalue],
                    ["Numéro de semaine ISO:", weekNumber],
                    ["Date d'export:", new Date().toLocaleDateString('fr-FR')],
                    [],
                    ["NOTE DE QUALITÉ"],
                    ["Note sur 100:", `${note.toFixed(1)}/100`],
                    ["Pourcentage de conformité:", `${note.toFixed(1)}%`],
                    ["Statut:", status],
                    [],
                    ["RÉSUMÉ DES RÉSULTATS"],
                    ["Statut", "Nombre", "Pourcentage"],
                    ["Conforme", counts.c, counts.total > 0 ? (counts.c / counts.total * 100).toFixed(1) + "%" : "0%"],
                    ["Non conforme", counts.nc, counts.total > 0 ? (counts.nc / counts.total * 100).toFixed(1) + "%" : "0%"],
                    ["Non applicable", counts.na, counts.total > 0 ? (counts.na / counts.total * 100).toFixed(1) + "%" : "0%"],
                    ["Non renseigné", counts.nr, counts.total > 0 ? (counts.nr / counts.total * 100).toFixed(1) + "%" : "0%"],
                    ["TOTAL", counts.total, "100%"],
                    [],
                    ["DÉTAILS PAR THÈME"],
                    ["Thème", "Conforme", "Non conforme", "Non applicable", "Non renseigné", "Total"]
                ];

                // Stats par thème
                const themeStats = {};
                CRITERIA_DATA.forEach(item => {
                    if (!themeStats[item.themeNumber]) {
                        themeStats[item.themeNumber] = {
                            theme: item.theme || `Thème ${item.themeNumber}`,
                            c: 0, nc: 0, na: 0, nr: 0, total: 0
                        };
                    }
                    const stat = themeStats[item.themeNumber];
                    stat.total++;
                    const status = complianceData[item.critere]?.status;
                    if (status === 'c') stat.c++;
                    else if (status === 'nc') stat.nc++;
                    else if (status === 'na') stat.na++;
                    else stat.nr++;
                });

                Object.values(themeStats).forEach(stat => {
                    summaryData.push([stat.theme, stat.c, stat.nc, stat.na, stat.nr, stat.total]);
                });

                // Création de la feuille
                const ws = XLSX.utils.aoa_to_sheet(summaryData);

                // Largeurs de colonnes
                ws['!cols'] = [
                    { wch: 40 }, { wch: 20 }, { wch: 15 }, 
                    { wch: 15 }, { wch: 15 }, { wch: 15 }
                ];

                // Fusions
                ws['!merges'] = [
                    { s: { r: 0, c: 0 }, e: { r: 0, c: 5 } },
                    { s: { r: 2, c: 0 }, e: { r: 2, c: 5 } },
                    { s: { r: 11, c: 0 }, e: { r: 11, c: 5 } },
                    { s: { r: 16, c: 0 }, e: { r: 16, c: 5 } },
                    { s: { r: 24, c: 0 }, e: { r: 24, c: 5 } }
                ];

                // Application des styles
                const range = XLSX.utils.decode_range(ws['!ref']);
                for (let R = range.s.r; R <= range.e.r; R++) {
                    for (let C = range.s.c; C <= range.e.c; C++) {
                        const cell = ws[XLSX.utils.encode_cell({ c: C, r: R })];
                        if (!cell) continue;

                        const style = { font: {}, fill: {}, alignment: {}, border: {} };

                        // Titre principal
                        if (R === 0) {
                            style.font = { sz: 18, bold: true, color: { rgb: "2C3E50" } };
                            style.alignment = { horizontal: "center" };
                        }
                        // En-têtes de sections
                        else if ([2, 11, 16, 24].includes(R)) {
                            style.font = { sz: 14, bold: true, color: { rgb: "FFFFFF" } };
                            style.fill = { fgColor: { rgb: "4A6491" } };
                        }
                        // En-têtes de tableaux
                        else if ([3, 12, 17, 25].includes(R)) {
                            style.font = { bold: true, color: { rgb: "FFFFFF" } };
                            style.fill = { fgColor: { rgb: "2C3E50" } };
                            style.border = {
                                top: { style: "thin", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            };
                        }
                        // Lignes de données
                        else if (R > 3) {
                            style.border = {
                                bottom: { style: "thin", color: { rgb: "CCCCCC" } },
                                left: { style: "thin", color: { rgb: "CCCCCC" } },
                                right: { style: "thin", color: { rgb: "CCCCCC" } }
                            };

                            // Couleur conditionnelle pour la note
                            if (R === 13 && C === 1) {
                                style.font = { bold: true, color: { rgb: note >= 85 ? "27ae60" : "e74c3c" } };
                            }
                            // Couleur conditionnelle pour le statut
                            if (R === 15 && C === 1) {
                                style.font = { bold: true, color: { rgb: note >= 85 ? "27ae60" : "e74c3c" } };
                            }
                        }

                        cell.s = style;
                    }
                }

                XLSX.utils.book_append_sheet(wb, ws, "Résumé");
            }

            function exportToExcel() {
                // Vérifications
                const docName = document.getElementById('doc-name').value;
                if (!docName) {
                    alert('Veuillez saisir le nom du document');
                    document.getElementById('doc-name').focus();
                    return;
                }

                for (const [critere, data] of Object.entries(complianceData)) {
                    if (data.status === 'nc' && !data.comment?.trim()) {
                        alert(`Le critère ${critere} est "Non conforme" sans commentaire`);
                        const row = document.querySelector(`[data-critere="${critere}"]`);
                        if (row) {
                            row.scrollIntoView({ behavior: 'smooth', block: 'center' });
                            row.querySelector('.comment-input').classList.add('show').focus();
                        }
                        return;
                    }
                }

                // Création du classeur
                document.getElementById('loading').style.display = 'block';
                const wb = XLSX.utils.book_new();

                // Feuille Conformité
                const wsData = [["Thème", "Critère", "Intitulé", "Conformité", "Commentaire"]];
                const mergeRanges = [];
                let currentTheme = "";
                let currentThemeRow = 2;

                CRITERIA_DATA.forEach((item, index) => {
                    const data = complianceData[item.critere] || {};
                    const conformite = data.status === 'c' ? "Conforme" : 
                                      data.status === 'nc' ? "Non conforme" : 
                                      data.status === 'na' ? "Non applicable" : "Non renseigné";

                    if (item.theme && item.theme !== currentTheme) {
                        currentTheme = item.theme;
                        currentThemeRow = wsData.length + 1;
                        wsData.push([currentTheme, item.critere, item.intitule, conformite, data.comment || ""]);
                    } else {
                        wsData.push(["", item.critere, item.intitule, conformite, data.comment || ""]);
                    }

                    if (index === CRITERIA_DATA.length - 1 || 
                        (CRITERIA_DATA[index + 1] && CRITERIA_DATA[index + 1].theme)) {
                        if (currentTheme && wsData.length > currentThemeRow) {
                            mergeRanges.push({
                                s: { r: currentThemeRow - 1, c: 0 },
                                e: { r: wsData.length - 1, c: 0 }
                            });
                        }
                    }
                });

                const ws = XLSX.utils.aoa_to_sheet(wsData);
                ws['!cols'] = [{ wch: 30 }, { wch: 10 }, { wch: 80 }, { wch: 15 }, { wch: 50 }];
                ws['!merges'] = mergeRanges;

                // Styles pour la feuille Conformité
                const range = XLSX.utils.decode_range(ws['!ref']);
                for (let R = range.s.r; R <= range.e.r; R++) {
                    for (let C = range.s.c; C <= range.e.c; C++) {
                        const cell = ws[XLSX.utils.encode_cell({ c: C, r: R })];
                        if (!cell) continue;

                        const style = { font: {}, fill: {}, alignment: {}, border: {} };

                        if (R === 0) {
                            style.font = { bold: true, color: { rgb: "FFFFFF" } };
                            style.fill = { fgColor: { rgb: "2C3E50" } };
                            style.alignment = { horizontal: "center", vertical: "center" };
                            style.border = {
                                top: { style: "thin", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            };
                        }
                        else if (R >= 1 && cell.v && cell.v.toString().match(/^\d+\.\s/)) {
                            style.font = { bold: true, sz: 12, color: { rgb: "000000" } };
                            style.alignment = { horizontal: "center", vertical: "center" };
                            style.border = {
                                top: { style: "medium", color: { rgb: "000000" } },
                                bottom: { style: "thin", color: { rgb: "000000" } },
                                left: { style: "thin", color: { rgb: "000000" } },
                                right: { style: "thin", color: { rgb: "000000" } }
                            };
                            const themeNum = parseInt(cell.v.split('.')[0]);
                            if (themeNum >= 1 && themeNum <= 10) {
                                style.fill = { fgColor: { rgb: THEME_COLORS[themeNum - 1] } };
                            }
                        }
                        else if (R > 0 && C <= 4) {
                            style.border = {
                                top: { style: "thin", color: { rgb: "CCCCCC" } },
                                bottom: { style: "thin", color: { rgb: "CCCCCC" } },
                                left: { style: "thin", color: { rgb: "CCCCCC" } },
                                right: { style: "thin", color: { rgb: "CCCCCC" } }
                            };
                            style.alignment = { vertical: "top", wrapText: true };

                            if (C === 3) {
                                style.alignment.horizontal = "center";
                                if (cell.v === "Conforme") style.font = { bold: true, color: { rgb: "006400" } };
                                else if (cell.v === "Non conforme") style.font = { bold: true, color: { rgb: "8B0000" } };
                                else if (cell.v === "Non applicable") style.font = { color: { rgb: "696969" } };
                            }
                        }

                        cell.s = style;
                    }
                }

                XLSX.utils.book_append_sheet(wb, ws, "Conformité");
                createSummarySheet(wb);

                // Export
                setTimeout(() => {
                    const wbout = XLSX.write(wb, { bookType: 'xlsx', type: 'binary' });
                    const buf = new ArrayBuffer(wbout.length);
                    const view = new Uint8Array(buf);
                    for (let i = 0; i < wbout.length; i++) view[i] = wbout.charCodeAt(i) & 0xFF;

                    const blob = new Blob([buf], { type: "application/octet-stream" });
                    const link = document.createElement("a");
                    link.href = URL.createObjectURL(blob);
                    link.download = `RAPDF_${docName.replace(/[^a-z0-9]/gi, '_').substring(0, 50)}_${new Date().toISOString().slice(0, 10)}.xlsx`;
                    link.click();

                    document.getElementById('loading').style.display = 'none';
                    alert(`Export réussi !\nNote: ${calculateQualityNote().toFixed(1)}/100`);
                }, 500);
            }

            // ==================== INITIALISATION ====================
            function init() {
                // Date du jour
                document.getElementById('current-date').textContent = 
                    new Date().toLocaleDateString('fr-FR', { day: '2-digit', month: '2-digit', year: 'numeric' });
                
                const today = new Date();
                document.getElementById('doc-date').valueAsDate = today;
                updateWeekNumber();

                // Génération du tableau
                generateComplianceTable();

                // Initialisation des données
                CRITERIA_DATA.forEach(item => {
                    complianceData[item.critere] = { status: "", comment: "", themeNumber: item.themeNumber };
                });

                // Chargement des données sauvegardées
                loadFromStorage();

                // Mise à jour de l'interface
                updateStatusSummary();
                updateQualityNoteDisplay();

                // Écouteurs d'événements
                document.addEventListener('change', e => {
                    if (e.target.classList.contains('compliance-status')) {
                        const row = e.target.closest('tr');
                        const critere = row.dataset.critere;
                        const status = e.target.value;
                        const commentInput = row.querySelector('.comment-input');

                        complianceData[critere].status = status;

                        if (status === 'nc') {
                            commentInput.classList.add('show');
                        } else {
                            commentInput.classList.remove('show');
                            commentInput.value = '';
                            complianceData[critere].comment = '';
                        }

                        updateStatusSummary();
                        updateQualityNoteDisplay();
                    }
                });

                document.addEventListener('input', e => {
                    if (e.target.classList.contains('comment-input')) {
                        const row = e.target.closest('tr');
                        const critere = row.dataset.critere;
                        complianceData[critere].comment = e.target.value;
                    }
                });

                document.getElementById('doc-date').addEventListener('change', updateWeekNumber);

                document.getElementById('export-btn').addEventListener('click', exportToExcel);
                
                document.getElementById('reset-btn').addEventListener('click', () => {
                    if (confirm('Réinitialiser toutes les données ?')) {
                        document.getElementById('doc-name').value = '';
                        document.getElementById('doc-type').value = '';
                        document.getElementById('evaluator').value = '';
                        document.getElementById('evalue').value = '';
                        document.getElementById('doc-date').valueAsDate = new Date();
                        updateWeekNumber();

                        complianceData = {};
                        CRITERIA_DATA.forEach(item => {
                            complianceData[item.critere] = { status: "", comment: "", themeNumber: item.themeNumber };
                        });

                        document.querySelectorAll('.compliance-status').forEach(s => s.value = '');
                        document.querySelectorAll('.comment-input').forEach(i => {
                            i.value = '';
                            i.classList.remove('show');
                        });

                        localStorage.removeItem('accessibilityComplianceData');
                        updateStatusSummary();
                        updateQualityNoteDisplay();
                        alert('Formulaire réinitialisé');
                    }
                });

                document.getElementById('save-btn').addEventListener('click', () => {
                    if (saveToStorage()) alert('Données sauvegardées');
                });
            }

            // Démarrage
            document.addEventListener('DOMContentLoaded', init);
        })();
