' UserForm Code (Name: frmFileManager)
' Create a new UserForm with this layout:
' - Label1: "Source Directory:"
' - TextBox1: txtSourcePath (Multiline = False, Width = 400)
' - CommandButton1: btnBrowse (Caption = "Browse...")
' - ListBox1: lstFiles (Width = 500, Height = 150)
' - Frame1: Caption = "Step 1: Initial Move (A TRAITER → EN COURS)"
'   - CommandButton2: btnToEnCours (Caption = "Move to EN COURS")
' - Frame2: Caption = "Step 2: Final Move (Any Location → TRAITES/NON CONFORME)"
'   - OptionButton1: optToTraite (Caption = "To TRAITES")
'   - OptionButton2: optToNonConforme (Caption = "To NON CONFORME")
'   - CommandButton3: btnMoveFinal (Caption = "Move Selected Folder to Destination")
' - Frame3: Caption = "Files in EN COURS (for reference)"
'   - ListBox2: lstEnCoursFiles (Width = 500, Height = 100)
' - ListBox3: lstLog (Width = 500, Height = 100)
' - CommandButton4: btnRefresh (Caption = "Refresh All")

Option Explicit

Private basePath As String
Private currentFolder As String
Private selectedFile As String
Private selectedFolder As String

Private Sub UserForm_Initialize()
    basePath = "\\UF11-a03\eids-eaa-pdf-statiques\Accessibilité\"
    currentFolder = basePath
    txtSourcePath.Text = currentFolder
    LoadFiles
    LoadEnCoursFiles
    LoadTrackingData
End Sub

Private Sub btnBrowse_Click()
    Dim fd As FileDialog
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)
    
    With fd
        .Title = "Select a folder containing files to process"
        .InitialFileName = currentFolder
        If .Show = -1 Then
            currentFolder = .SelectedItems(1) & "\"
            txtSourcePath.Text = currentFolder
            LoadFiles
        End If
    End With
End Sub

Private Sub lstFiles_Change()
    If lstFiles.ListIndex >= 0 Then
        ' Remove [FOLDER] prefix if present
        selectedFile = Replace(lstFiles.List(lstFiles.ListIndex), "[FOLDER] ", "")
    End If
End Sub

Private Sub lstEnCoursFiles_Change()
    If lstEnCoursFiles.ListIndex >= 0 Then
        ' Get the selected folder from EN COURS list
        selectedFolder = Replace(lstEnCoursFiles.List(lstEnCoursFiles.ListIndex), "[FOLDER] ", "")
    End If
End Sub

Private Sub btnToEnCours_Click()
    If selectedFile = "" Then
        MsgBox "Please select a file or folder first.", vbExclamation
        Exit Sub
    End If
    
    Dim sourcePath As String
    Dim destPath As String
    Dim fileName As String
    Dim folderName As String
    
    ' Get the full source path
    sourcePath = currentFolder & selectedFile
    
    ' Check if it's a folder or file
    If (GetAttr(sourcePath) And vbDirectory) = vbDirectory Then
        ' It's a folder - get the folder name
        folderName = selectedFile
    Else
        ' It's a file - get the file name without extension for folder
        folderName = Left(selectedFile, InStrRev(selectedFile, ".") - 1)
    End If
    
    ' Extract the client name and subpath from the current folder
    Dim clientName As String
    Dim subPath As String
    Dim searchStr As String
    
    ' Find which client folder we're in
    searchStr = basePath
    Dim remainingPath As String
    remainingPath = Mid(currentFolder, Len(searchStr) + 1)
    
    ' Extract client name (first folder after basePath)
    Dim parts() As String
    parts = Split(remainingPath, "\")
    
    If UBound(parts) >= 0 Then
        clientName = parts(0)
        
        ' Build subpath after A TRAITER
        Dim aTraiterPos As Integer
        aTraiterPos = InStr(remainingPath, "A TRAITER\")
        
        If aTraiterPos > 0 Then
            subPath = Mid(remainingPath, aTraiterPos + Len("A TRAITER\"))
        Else
            subPath = ""
        End If
    Else
        MsgBox "Could not determine client folder structure.", vbExclamation
        Exit Sub
    End If
    
    ' Construct destination path in EN COURS with proper client folder
    If subPath <> "" Then
        destPath = basePath & clientName & "\EN COURS\" & subPath & folderName & "\"
    Else
        destPath = basePath & clientName & "\EN COURS\" & folderName & "\"
    End If
    
    ' Create destination folder if it doesn't exist
    On Error GoTo CreateFolderError
    If Dir(destPath, vbDirectory) = "" Then
        MkDir destPath
    End If
    
    ' Move the folder/file
    On Error GoTo MoveError
    If (GetAttr(sourcePath) And vbDirectory) = vbDirectory Then
        ' It's a folder - move all contents
        Dim fso As Object
        Set fso = CreateObject("Scripting.FileSystemObject")
        
        ' Move all files from source folder to destination
        Dim fileInFolder As String
        fileInFolder = Dir(sourcePath & "\*.*")
        Do While fileInFolder <> ""
            FileCopy sourcePath & "\" & fileInFolder, destPath & fileInFolder
            Kill sourcePath & "\" & fileInFolder
            fileInFolder = Dir()
        Loop
        
        ' Remove the empty source folder
        RmDir sourcePath
    Else
        ' It's a file - move the file
        FileCopy sourcePath, destPath & selectedFile
        Kill sourcePath
    End If
    
    ' Record the first pick (pick from A TRAITER)
    RecordFirstPick folderName, clientName & "\" & subPath
    
    ' Log the action
    lstLog.AddItem Format(Now, "yyyy-mm-dd hh:mm:ss") & " - Moved: " & selectedFile & " to EN COURS under " & clientName
    
    ' Refresh file lists
    LoadFiles
    LoadEnCoursFiles
    
    MsgBox "File moved successfully to EN COURS under " & clientName & "!", vbInformation
    Exit Sub
    
CreateFolderError:
    MsgBox "Error creating destination folder: " & Err.Description & vbCrLf & "Path: " & destPath, vbCritical
    Exit Sub
    
MoveError:
    MsgBox "Error moving file: " & Err.Description, vbCritical
End Sub

Private Sub btnMoveFinal_Click()
    If selectedFile = "" Then
        MsgBox "Please select a folder from the source location first.", vbExclamation
        Exit Sub
    End If
    
    If selectedFolder = "" Then
        MsgBox "Please select a matching folder from the EN COURS list.", vbExclamation
        Exit Sub
    End If
    
    ' Verify that the selected folder name matches
    If selectedFile <> selectedFolder Then
        MsgBox "The selected folder name must match the EN COURS folder name." & vbCrLf & _
               "Selected: " & selectedFile & vbCrLf & _
               "EN COURS folder: " & selectedFolder, vbExclamation
        Exit Sub
    End If
    
    If Not optToTraite And Not optToNonConforme Then
        MsgBox "Please select destination (TRAITES or NON CONFORME).", vbExclamation
        Exit Sub
    End If
    
    Dim destType As String
    If optToTraite Then
        destType = "TRAITES"
    Else
        destType = "NON CONFORME"
    End If
    
    ' Get the source folder path
    Dim sourceFolder As String
    sourceFolder = currentFolder & selectedFile
    
    ' Check if source is a folder
    If (GetAttr(sourceFolder) And vbDirectory) <> vbDirectory Then
        MsgBox "Please select a folder to move.", vbExclamation
        Exit Sub
    End If
    
    ' Find the matching EN COURS folder to get its path and delete it
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    Dim enCoursFolderPath As String
    Dim fullPath As String
    Dim found As Boolean
    
    ' Find the EN COURS folder path from tracking data
    Dim i As Long
    For i = 2 To lastRow
        If ws.Cells(i, 1).Value = selectedFolder And IsEmpty(ws.Cells(i, 5).Value) Then
            fullPath = ws.Cells(i, 2).Value
            found = True
            Exit For
        End If
    Next i
    
    If Not found Then
        MsgBox "Could not find matching folder in EN COURS tracking data.", vbExclamation
        Exit Sub
    End If
    
    ' Construct the EN COURS folder path
    Dim pathParts() As String
    pathParts = Split(fullPath, "\")
    
    If UBound(pathParts) >= 0 Then
        enCoursFolderPath = basePath & pathParts(0) & "\EN COURS\"
        
        ' Add subpath if exists
        If UBound(pathParts) >= 1 Then
            Dim j As Integer
            For j = 1 To UBound(pathParts)
                If pathParts(j) <> "" Then
                    enCoursFolderPath = enCoursFolderPath & pathParts(j) & "\"
                End If
            Next j
        End If
        
        enCoursFolderPath = enCoursFolderPath & selectedFolder & "\"
    Else
        MsgBox "Invalid path structure in tracking data.", vbExclamation
        Exit Sub
    End If
    
    ' Check if EN COURS folder exists
    If Dir(enCoursFolderPath, vbDirectory) = "" Then
        MsgBox "EN COURS folder not found at: " & enCoursFolderPath, vbExclamation
        Exit Sub
    End If
    
    ' Construct destination folder with proper client structure
    Dim destFolder As String
    destFolder = basePath & pathParts(0) & "\" & destType & "\"
    
    ' Add subpath if exists
    If UBound(pathParts) >= 1 Then
        Dim k As Integer
        For k = 1 To UBound(pathParts)
            If pathParts(k) <> "" Then
                destFolder = destFolder & pathParts(k) & "\"
            End If
        Next k
    End If
    
    destFolder = destFolder & selectedFolder & "\"
    
    ' Create destination folder structure
    On Error GoTo CreateDestError
    If Dir(destFolder, vbDirectory) = "" Then
        ' Create all necessary subfolders
        Dim folderParts() As String
        folderParts = Split(destFolder, "\")
        Dim buildPath As String
        buildPath = folderParts(0) & "\" & folderParts(1) & "\" & folderParts(2) & "\" & folderParts(3) & "\"
        
        Dim m As Integer
        For m = 4 To UBound(folderParts) - 1
            buildPath = buildPath & folderParts(m) & "\"
            If Dir(buildPath, vbDirectory) = "" Then
                MkDir buildPath
            End If
        Next m
        
        ' Create final folder
        MkDir destFolder
    End If
    
    ' Move files from source folder to destination
    On Error GoTo MoveError
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    
    ' Move all files from source folder to destination
    Dim fileInFolder As String
    fileInFolder = Dir(sourceFolder & "\*.*")
    Do While fileInFolder <> ""
        FileCopy sourceFolder & "\" & fileInFolder, destFolder & fileInFolder
        Kill sourceFolder & "\" & fileInFolder
        fileInFolder = Dir()
    Loop
    
    ' Remove the empty source folder
    RmDir sourceFolder
    
    ' Delete the EN COURS folder
    On Error Resume Next
    Kill enCoursFolderPath & "*.*"
    RmDir enCoursFolderPath
    On Error GoTo MoveError
    
    ' Record the second pick and calculate DMT
    RecordSecondPick selectedFolder, fullPath, destType
    
    ' Log the action
    lstLog.AddItem Format(Now, "yyyy-mm-dd hh:mm:ss") & " - Completed: " & selectedFolder & " moved to " & destType & " (EN COURS folder deleted)"
    
    ' Refresh file lists
    LoadFiles
    LoadEnCoursFiles
    
    MsgBox "Folder processed successfully!" & vbCrLf & _
           "Files moved to: " & destType & vbCrLf & _
           "EN COURS folder has been deleted.", vbInformation
    Exit Sub
    
CreateDestError:
    MsgBox "Error creating destination folder: " & Err.Description & vbCrLf & "Path: " & destFolder, vbCritical
    Exit Sub
    
MoveError:
    MsgBox "Error moving folder: " & Err.Description, vbCritical
End Sub

Private Sub btnRefresh_Click()
    LoadFiles
    LoadEnCoursFiles
    LoadTrackingData
End Sub

Private Sub LoadFiles()
    lstFiles.Clear
    
    Dim folder As String
    folder = txtSourcePath.Text
    
    If Dir(folder, vbDirectory) = "" Then
        Exit Sub
    End If
    
    ' List folders only (since we need to select folders for step 2)
    Dim fileName As String
    fileName = Dir(folder & "*", vbDirectory)
    
    Do While fileName <> ""
        If fileName <> "." And fileName <> ".." Then
            If (GetAttr(folder & fileName) And vbDirectory) = vbDirectory Then
                lstFiles.AddItem "[FOLDER] " & fileName
            End If
        End If
        fileName = Dir()
    Loop
    
    ' Reset selected file
    selectedFile = ""
End Sub

Private Sub LoadEnCoursFiles()
    lstEnCoursFiles.Clear
    
    ' Get all folders currently in EN COURS from tracking data
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If lastRow > 1 Then
        Dim i As Long
        For i = 2 To lastRow
            ' Only show items that are in EN COURS (have first pick but no second pick)
            If Not IsEmpty(ws.Cells(i, 3).Value) And IsEmpty(ws.Cells(i, 5).Value) Then
                lstEnCoursFiles.AddItem "[FOLDER] " & ws.Cells(i, 1).Value & " (" & ws.Cells(i, 2).Value & ")"
            End If
        Next i
    End If
    
    If lstEnCoursFiles.ListCount = 0 Then
        lstEnCoursFiles.AddItem "No folders in EN COURS"
    End If
    
    ' Reset selected folder
    selectedFolder = ""
End Sub

Private Sub RecordFirstPick(folderName As String, fullPath As String)
    ' Store in a worksheet for tracking
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    ' Find the next empty row
    Dim nextRow As Long
    nextRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    
    ' Record the pick
    ws.Cells(nextRow, 1).Value = folderName  ' File/Folder name
    ws.Cells(nextRow, 2).Value = fullPath    ' Full path (client + subfolders)
    ws.Cells(nextRow, 3).Value = Now         ' First pick date/time
    ws.Cells(nextRow, 4).Value = Environ("username") ' User/Agent
    ws.Cells(nextRow, 7).Value = ThisWorkbook.Name ' Session/Workbook name
End Sub

Private Sub RecordSecondPick(folderName As String, fullPath As String, destType As String)
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    ' Find the matching record (folder with same name and path without second pick)
    Dim i As Long
    For i = 2 To lastRow
        If ws.Cells(i, 1).Value = folderName And _
           ws.Cells(i, 2).Value = fullPath And _
           IsEmpty(ws.Cells(i, 5).Value) Then
            
            ' Record second pick
            ws.Cells(i, 5).Value = Now  ' Second pick date/time
            ws.Cells(i, 6).Value = destType ' Destination
            
            ' Calculate DMT (in minutes then convert to time format)
            Dim dmtMinutes As Double
            dmtMinutes = DateDiff("n", ws.Cells(i, 3).Value, Now) ' Minutes difference
            ws.Cells(i, 8).Value = dmtMinutes / 1440 ' Convert to days (for Excel time format)
            ws.Cells(i, 8).NumberFormat = "[h]:mm:ss" ' Format as hours:minutes:seconds
            
            Exit For
        End If
    Next i
End Sub

Private Function GetTrackingWorksheet() As Worksheet
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim trackingSheetName As String
    trackingSheetName = "Tracking"
    
    ' Check if tracking worksheet exists in current workbook
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(trackingSheetName)
    On Error GoTo 0
    
    If ws Is Nothing Then
        ' Create tracking worksheet
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = trackingSheetName
        
        ' Add headers
        With ws
            .Cells(1, 1).Value = "File/Folder Name"
            .Cells(1, 2).Value = "Full Path (Client\Subfolders)"
            .Cells(1, 3).Value = "First Pick (A TRAITER to EN COURS)"
            .Cells(1, 4).Value = "Agent"
            .Cells(1, 5).Value = "Second Pick (EN COURS to Final)"
            .Cells(1, 6).Value = "Final Destination"
            .Cells(1, 7).Value = "Session/Workbook"
            .Cells(1, 8).Value = "DMT (Processing Time)"
            
            ' Format headers
            .Range("A1:H1").Font.Bold = True
            .Range("A1:H1").Interior.Color = RGB(200, 200, 200)
            .Columns("A:H").AutoFit
        End With
    End If
    
    Set GetTrackingWorksheet = ws
End Function

Private Sub LoadTrackingData()
    ' Load recent tracking data into log listbox
    Dim ws As Worksheet
    Set ws = GetTrackingWorksheet
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If lastRow > 1 Then
        lstLog.Clear
        lstLog.AddItem "=== Recent Activity ==="
        
        Dim i As Long
        Dim startRow As Long
        startRow = Application.WorksheetFunction.Max(2, lastRow - 9) ' Show last 10 entries
        
        For i = lastRow To startRow Step -1
            Dim logEntry As String
            logEntry = Format(ws.Cells(i, 3).Value, "yyyy-mm-dd hh:mm") & " - " & ws.Cells(i, 1).Value & " (" & ws.Cells(i, 2).Value & ") picked by " & ws.Cells(i, 4).Value
            If Not IsEmpty(ws.Cells(i, 5).Value) Then
                logEntry = logEntry & " → Completed in " & Format(ws.Cells(i, 8).Value, "[h]:mm:ss")
            Else
                logEntry = logEntry & " → In Progress"
            End If
            lstLog.AddItem logEntry
        Next i
    End If
End Sub
