---

Lister fichiers

Public Sub GetFiles(path As String, lv As ListView)

Dim folder As Folder
Dim file As File
Dim ext As String
Dim item As ListItem

lv.ListItems.Clear

Set folder = fso.GetFolder(path)

For Each file In folder.Files

    ext = LCase(fso.GetExtensionName(file.Name))

    If IsAllowed(ext) Then
    
        If Left(file.Name, 2) <> "~$" Then
        
            Set item = lv.ListItems.Add(, , file.Name)
            
            item.SubItems(1) = ext
            item.SubItems(2) = file.DateLastModified
            item.SubItems(3) = Round(file.Size / 1024, 2) & " KB"
        
        End If
        
    End If

Next

End Sub


---

Filtre extensions

Public Function IsAllowed(ext As String) As Boolean

Dim i

For i = LBound(AllowedExt) To UBound(AllowedExt)

    If ext = AllowedExt(i) Then
    
        IsAllowed = True
        Exit Function
        
    End If

Next

End Function

📁 Module modLog

Public Sub WriteLog(action As String, src As String, dest As String, result As String)

Dim ws As Worksheet
Dim r As Long

Set ws = ThisWorkbook.Sheets("LOG")

r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1

ws.Cells(r, 1) = Now
ws.Cells(r, 2) = Environ("Username")
ws.Cells(r, 3) = action
ws.Cells(r, 4) = src
ws.Cells(r, 5) = dest
ws.Cells(r, 6) = result

End Sub

🗑 Supprimer fichier

Private Sub cmdDelete_Click()

Dim filePath

filePath = tvFolders.SelectedItem.Key & "\" & lvFiles.SelectedItem.Text

If MsgBox("Supprimer ce fichier ?", vbYesNo) = vbNo Then Exit Sub

If Not IsSafePath(filePath) Then

MsgBox "Action refusée"

Exit Sub

End If

fso.DeleteFile filePath, True

WriteLog "DELETE", filePath, "", "OK"

GetFiles tvFolders.SelectedItem.Key, lvFiles

End Sub

✅ Marquer fichier traité

Private Sub cmdTraite_Click()

Dim src
Dim destFolder
Dim dest

src = tvFolders.SelectedItem.Key & "\" & lvFiles.SelectedItem.Text

destFolder = PATH_TRAITES & Format(Date, "yyyy-mm") & "\"

If Not fso.FolderExists(destFolder) Then

MkDir destFolder

End If

dest = destFolder & lvFiles.SelectedItem.Text

fso.MoveFile src, dest

WriteLog "MOVE", src, dest, "OK"

GetFiles tvFolders.SelectedItem.Key, lvFiles

End Sub
